33 CCylinder::CCylinder(const CCylinder& rhs, Compartment* c)
34  : _compartment(c), _pCylinder(rhs._pCylinder), _size(rhs._size) {
35 
36  CCylinder* rhsPtr = const_cast<CCylinder*>(&rhs);
37 
38  //copy all monomers, bounds
39  for(auto &m : rhs._monomers)
40  _monomers.emplace_back(m->clone(c));
41 
42  //copy all internal reactions
43  for(auto &r: rhs._internalReactions) {
44  ReactionBase* rxnClone = r->clone(c->getSpeciesContainer());
45 
46  if(r->getCBound() != nullptr)
47  r->getCBound()->setOffReaction(rxnClone);
48 
49  addInternalReaction(rxnClone);
50  }
51  //copy all cross-cylinder reactions
52  for(auto it = rhs._crossCylinderReactions.begin();
53  it != rhs._crossCylinderReactions.end(); it++) {
54 
55  for(auto &r : it->second) {
56 
57  //copy cbound if any
58  ReactionBase* rxnClone = r->clone(c->getSpeciesContainer());
59 
60  if(r->getCBound() != nullptr)
61  r->getCBound()->setOffReaction(rxnClone);
62 
63  addCrossCylinderReaction(it->first, rxnClone);
64  }
65  }
66  //Copy reacting cylinders, Clone reactions where this cylinder is involved
67  for(auto &ccyl : rhs._reactingCylinders) {
68 
69  //clone reactions
70  for(auto &r: ccyl->getCrossCylinderReactions()[rhsPtr]) {
71 
72  //copy cbound if any
73  ReactionBase* rxnClone = r->clone(c->getSpeciesContainer());
74 
75  if(r->getCBound() != nullptr)
76  r->getCBound()->setOffReaction(rxnClone);
77 
78  ccyl->addCrossCylinderReaction(this, rxnClone);
79  }
80  }
81 }

138 void CCylinder::removeCrossCylinderReaction(CCylinder* other,
139  ReactionBase* r) {
140 
141  auto it = _crossCylinderReactions[other].find(r);
142  if(it != _crossCylinderReactions[other].end()) {
143 
144  //erase the reaction
145  _crossCylinderReactions[other].erase(it);
146 
147  //passivate reaction, removing from dependents
148  r->passivateReaction();
149 
150  //remove from compartment and chemsim
151  _chemSim->removeReaction(r);
152  _compartment->removeInternalReaction(r);
153 
154  //if number of reactions in cross-cylinder
155  //has dropped to zero, delete it
156  if(_crossCylinderReactions[other].empty()) {
157 
158  _crossCylinderReactions.erase(other);
159 
160  //also remove from reacting of other ccylinder
161  auto it2 =other->_reactingCylinders.find(this);
162 
163  if(it2 != other->_reactingCylinders.end())
164  other->_reactingCylinders.erase(it2);
165  }
