29 void CylinderCylinderNL::updateNeighbors(Cylinder* cylinder, bool runtime) {
30 
31  //clear existing
32  _list[cylinder].clear();
33 
34  //Find surrounding compartments (For now its conservative)
35  vector<Compartment*> compartments;
36  auto searchDist = SysParams::Geometry().largestCompartmentSide;
37 
38  GController::findCompartments(cylinder->coordinate,
39  cylinder->getCompartment(),
40  searchDist + _rMax, compartments);
41 
42  for(auto &comp : compartments) {
43  for(auto &ncylinder : comp->getCylinders()) {
44 
45  //Don't add the same cylinder!
46  if(cylinder == ncylinder) continue;
47 
48  //Dont add if ID is more than cylinder for half-list
49  if(!_full && cylinder->getID() <= ncylinder->getID()) continue;
50 
51  //Don't add if belonging to same parent
52  if(cylinder->getParent() == ncylinder->getParent()) {
53 
54  //if not cross filament, check if not neighboring
55  auto dist = fabs(cylinder->getPosition() -
56  ncylinder->getPosition());
57  if(dist <= 2) continue;
58  }
59 
60  //Dont add if not within range
61  double dist = twoPointDistance(cylinder->coordinate,
62  ncylinder->coordinate);
63  if(dist > _rMax || dist < _rMin) continue;
64 
65  //If we got through all of this, add it!
66  _list[cylinder].push_back(ncylinder);
67 
68  //if runtime, add to other list as well if full
69  if(runtime && _full) _list[ncylinder].push_back(cylinder);
70  }
71  }
72 }

86  Cylinder* cylinder;

88 
89  _list.erase(cylinder);
90 
91  //remove from other lists
