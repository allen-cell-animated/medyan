353 struct BranchingCallback {
354 
355  SubSystem* _ps;
356 
357  BranchingManager* _bManager;
358 
359  short _plusEnd;
360

362  float _offRate;
363

365  float onRate, float offRate, SubSystem* ps)
366

368  _plusEnd(plusEnd), _onRate(onRate), _offRate(offRate) {}
369 
370  void operator() (ReactionBase *r) {
371  BranchingPoint* b;
372  float frate;
373  short branchType = _bManager->getBoundInt();
374 
375  //choose a random binding site from manager
376  auto site = _bManager->chooseBindingSite();
377 
378  //get info from site
379  Cylinder* c1 = get<0>(site)->getCylinder();
380  short filType = c1->getType();
381 
382  double pos = double(get<1>(site)) / SysParams::Geometry().cylinderNumMon[filType];
383  if(SysParams::RUNSTATE==true){
384  //Get a position and direction of a new filament
385  auto x1 = c1->getFirstBead()->coordinate;
386  auto x2 = c1->getSecondBead()->coordinate;
387 
388  //get original direction of cylinder
389  auto p= midPointCoordinate(x1, x2, pos);
390  vector<double> n = twoPointDirection(x1, x2);
391 
392  //get branch projection
393 #ifdef MECHANICS
394  //use mechanical parameters
395  double l, t;
396  if(SysParams::Mechanics().BrStretchingL.size() != 0) {
397  l = SysParams::Mechanics().BrStretchingL[branchType];
398  t = SysParams::Mechanics().BrBendingTheta[branchType];
399  }
400  else {
401  cout << "Branching initialization cannot occur unless mechanical parameters are specified."
402  << " Using default values for Arp2/3 complex - l=10.0nm, theta=70.7deg"
403  << endl;
404  l = 10.0;
405  t = 1.22;
406  }
407 #else
408  cout << "Branching initialization cannot occur unless mechanics is enabled. Using"
409  << " default values for Arp2/3 complex - l=10.0nm, theta=70.7deg"
410  << endl;
411  double l = 10.0;
412  double t = 1.22;
413 #endif
414  double s = SysParams::Geometry().monomerSize[filType];
415 
416  auto branchPosDir = branchProjection(n, p, l, s, t);
417  auto bd = get<0>(branchPosDir); auto bp = get<1>(branchPosDir);
418 
419  //create a new filament
420  Filament* f = _ps->addTrackable<Filament>(_ps, filType, bp, bd, true, true);
421 
422  //mark first cylinder
423  Cylinder* c = f->getCylinderVector().front();
424  c->getCCylinder()->getCMonomer(0)->speciesPlusEnd(_plusEnd)->up();
425 
426  //create new branch
427  b= _ps->addTrackable<BranchingPoint>(c1, c, branchType, pos);
428  frate=_offRate;
429  }
430  else
431  {
432  CCylinder* c;
433  vector<tuple<tuple<CCylinder*, short>, tuple<CCylinder*, short>>> BrT=_bManager->getbtuple();
434  for(auto T:BrT){
435  CCylinder* cx=get<0>(get<0>(T));
436  double p = double(get<1>(get<0>(T)))/ double(SysParams::Geometry().cylinderNumMon[filType]);
437  if(cx->getCylinder()->getID()==c1->getID() && p==pos){
438  c=get<0>(get<1>(T));
439  break;
440  }}
441  b= _ps->addTrackable<BranchingPoint>(c1, c->getCylinder(), branchType, pos);
442  CMonomer* x=c->getCMonomer(0);
443  x->speciesMinusEnd(0)->down();
444  x->speciesFilament(0)->up();
445  frate=0.0;
446  }
447 
448  //create off reaction
449  auto cBrancher = b->getCBranchingPoint();
450 
451  cBrancher->setRates(_onRate, frate);
452  cBrancher->createOffReaction(r, _ps);
453  cBrancher->getOffReaction()->setBareRate(SysParams::BUBBareRate[branchType]);
454  }
