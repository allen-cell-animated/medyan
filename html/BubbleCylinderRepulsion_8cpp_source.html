27 template <class BRepulsionInteractionType>
28 double BubbleCylinderRepulsion<BRepulsionInteractionType>::computeEnergy(double d) {
29 
30  double U = 0;
31  double U_i;
32 
33  for (auto bb: Bubble::getBubbles()) {
34 
35  for(auto &c : _neighborList->getNeighbors(bb)) {
36 
37  //if part of an MTOC, skip
38  if(bb->isMTOC()) {
39 
40  auto mtoc = (MTOC*)bb->getParent();
41  auto filaments = mtoc->getFilaments();
42 
43  auto f = (Filament*)c->getParent();
44 
45  if(find(filaments.begin(), filaments.end(), f) != filaments.end())
46  continue;
47  }
48 
49  double kRep = bb->getRepulsionConst();
50  double screenLength = bb->getScreeningLength();
51 
52  double radius = bb->getRadius();
53 
54  Bead* bd1 = bb->getBead();
55 
56  //potential acts on second bead unless this is a minus end
57  Bead* bd2;
58  if(c->isMinusEnd())
59  bd2 = c->getFirstBead();
60  else
61  bd2 = c->getSecondBead();
62 
63  if (d == 0.0)
64  U_i = _FFType.energy(bd1, bd2, radius, kRep, screenLength);
65  else
66  U_i = _FFType.energy(bd1, bd2, radius, kRep, screenLength, d);
67 
68  if(fabs(U_i) == numeric_limits<double>::infinity()
69  || U_i != U_i || U_i < -1.0) {
70 
71  //set culprits and return
72  _otherCulprit = c;
73  _bubbleCulprit = bb;
74 
75  return -1;
76  }
77  else
78  U += U_i;
79  }
80  }
81 
82  return U;
83 }
