216 void Controller::setupInitialNetwork(SystemParser& p) {
217 
218  //Read bubble setup, parse bubble input file if needed
219  BubbleSetup BSetup = p.readBubbleSetup();
220  BubbleData bubbles;
221 
222  cout << "---" << endl;
223  cout << "Initializing bubbles...";
224 
225  if(BSetup.inputFile != "") {
226  BubbleParser bp(_inputDirectory + BSetup.inputFile);
227  bubbles = bp.readBubbles();
228  }
229  //add other bubbles if specified
230  BubbleInitializer* bInit = new RandomBubbleDist();
231 
232  auto bubblesGen = bInit->createBubbles(_subSystem->getBoundary(),
233  BSetup.numBubbles,
234  BSetup.bubbleType);
235  bubbles.insert(bubbles.end(), bubblesGen.begin(), bubblesGen.end());
236  delete bInit;
237 
238  //add bubbles
239  for (auto it: bubbles) {
240 
241  auto coord = get<1>(it);
242  auto type = get<0>(it);
243 
244  if(type >= SysParams::Mechanics().numBubbleTypes) {
245  cout << "Bubble data specified contains an "
246  <<"invalid bubble type. Exiting." << endl;
247  exit(EXIT_FAILURE);
248  }
249  _subSystem->addTrackable<Bubble>(_subSystem, coord, type);
250  }
251  cout << "Done. " << bubbles.size() << " bubbles created." << endl;
252 
253  //Read filament setup, parse filament input file if needed
254  FilamentSetup FSetup = p.readFilamentSetup();
255 // FilamentData filaments;
256 
257  cout << "---" << endl;
258  cout << "Initializing filaments...";
259 
260  if(FSetup.inputFile != "") {
261  FilamentParser fp(_inputDirectory + FSetup.inputFile);
262  filaments = fp.readFilaments();
263  }
264  fil=get<0>(filaments);
265  //add other filaments if specified
266  FilamentInitializer* fInit = new RandomFilamentDist();
267 
268  auto filamentsGen = fInit->createFilaments(_subSystem->getBoundary(),
269  FSetup.numFilaments,
270  FSetup.filamentType,
271  FSetup.filamentLength);
272  auto filGen=get<0>(filamentsGen);
273  fil.insert(fil.end(), filGen.begin(), filGen.end());
274  delete fInit;
275 
276  //add filaments
277  for (auto it: fil) {
278 
279  auto coord1 = get<1>(it);
280  auto coord2 = get<2>(it);
281  auto type = get<0>(it);
282 
283  if(type >= SysParams::Chemistry().numFilaments) {
284  cout << "Filament data specified contains an "
285  <<"invalid filament type. Exiting." << endl;
286  exit(EXIT_FAILURE);
287  }
288  vector<vector<double>> coords = {coord1, coord2};
289  if(coord2.size()==3){
290 
291  double d = twoPointDistance(coord1, coord2);
292  vector<double> tau = twoPointDirection(coord1, coord2);
293  int numSegment = d / SysParams::Geometry().cylinderSize[type];
294  // check how many segments can fit between end-to-end of the filament
295  if (numSegment == 0)
296  _subSystem->addTrackable<Filament>(_subSystem, type, coords, 2, FSetup.projectionType);
297  else
298  _subSystem->addTrackable<Filament>(_subSystem, type, coords, numSegment + 1, FSetup.projectionType);
299  }
300  else if(coord2.size()>3){
301  int numSegment = coord2.size()/3;
302  vector<vector<double>> coords;
303  coords.push_back(coord1);
304  for(int id=0;id<numSegment;id++)
305  coords.push_back({coord2[id*3],coord2[id*3+1],coord2[id*3+2]});
306 
307  if (numSegment == 0)
308  _subSystem->addTrackable<Filament>(_subSystem, type, coords, 2, FSetup.projectionType);
309  else
310  _subSystem->addTrackable<Filament>(_subSystem, type, coords, numSegment + 1, FSetup.projectionType);
311  }
312  }
313  cout << "Done. " << fil.size() << " filaments created." << endl;
314 }

370 void Controller::moveBoundary(double deltaTau) {
371 
372  //calculate distance to move
373  double dist = SysParams::Boundaries().moveSpeed * deltaTau;
374 
375  //move it
376  if(tau() >= SysParams::Boundaries().moveStartTime &&
377  tau() <= SysParams::Boundaries().moveEndTime)
378  _subSystem->getBoundary()->move(dist);
379 
380  //activate, deactivate necessary compartments
381  for(auto C : _subSystem->getCompartmentGrid()->getCompartments()) {
382 
383  if(_subSystem->getBoundary()->within(C)) {
384 
385  if(C->isActivated()) continue;
386  else _cController->activate(C);
387  }
388  else {
389  if(!C->isActivated()) continue;
390  else _cController->deactivate(C);
391  }
392  }

404  }
405 
406  //making linkers static
407  if(SysParams::Chemistry().makeLinkersStatic &&

409 
410  // loop through all linkers, passivate unbinding
411  for(auto l: Linker::getLinkers())
412  l->getCLinker()->getOffReaction()->passivateReaction();
413  }
414 
415 
416  if(SysParams::Mechanics().pinBoundaryFilaments &&
417  tau() >= SysParams::Mechanics().pinTime) {
418 
419  pinBoundaryFilaments();
420  }

481 void Controller::run() {
482 
483 #ifdef CHEMISTRY
484  double tauLastSnapshot = 0;
485  double tauLastMinimization = 0;
486  double tauLastNeighborList = 0;
487  double oldTau = 0;
488 
489  long stepsLastSnapshot = 0;
490  long stepsLastMinimization = 0;
491  long stepsLastNeighborList = 0;
492 
493  long totalSteps = 0;
494 #endif
495  chrono::high_resolution_clock::time_point chk1, chk2;
496  chk1 = chrono::high_resolution_clock::now();
497 //RESTART PHASE BEGINS
498  if(SysParams::RUNSTATE==false){
499  cout<<"RESTART PHASE BEINGS."<<endl;
500  Restart* _restart = new Restart(_subSystem, filaments,_chemData);
501 //Step 1. Turn off diffusion, passivate filament reactions and empty binding managers.
502  _restart->settorestartphase();
503  cout<<"Turned off Diffusion, filament reactions."<<endl;
504 //Step 2. Add bound species to their respective binding managers. Turn off unbinding, update propensities.
505  _restart->addtoHeaplinkermotor();
506  _restart->addtoHeapbranchers();
507  cout<<"Bound species added to reaction heap."<<endl;
508 //Step 3. ############ RUN LINKER/MOTOR REACTIONS TO BIND BRANCHERS, LINKERS, MOTORS AT RESPECTIVE POSITIONS.#######
509  std::cout<<"Reactions to be fired "<<_restart->getnumchemsteps()<<endl;
510  _cController->runSteps(_restart->getnumchemsteps());
511  cout<<"Reactions fired! Displaying heap"<<endl;
512 //Step 4. Display the number of reactions yet to be fired. Should be zero.
513  for(auto C : _subSystem->getCompartmentGrid()->getCompartments()) {
514  for(auto &Mgr:C->getFilamentBindingManagers()){cout<< Mgr->numBindingSites()<<' ';}}
515  cout<<endl;
516  _restart->redistributediffusingspecies();
517  cout<<"Diffusion rates restored, diffusing molecules redistributed."<<endl;
518 //Step 5. run mcontroller, update system, turn off restart state.
519  cout<<"Minimizing energy"<<endl;
520  _mController->run(false);
521  SysParams::RUNSTATE=true;
522  //reupdate positions and neighbor lists
523  updatePositions();
524  updateNeighborLists();
525 
526 //Step 6. Set Off rates back to original value.
527  for(auto LL : Linker::getLinkers())
528  {
529  LL->getCLinker()->setOffRate(LL->getCLinker()->getOffReaction()->getBareRate());
530  LL->getCLinker()->getOffReaction()->setRate(LL->getCLinker()->getOffReaction()->getBareRate());
531  LL->updateReactionRates();
532  LL->getCLinker()->getOffReaction()->updatePropensity();
533 
534  }
535  for(auto MM : MotorGhost::getMotorGhosts())
536  {
537  MM->getCMotorGhost()->setOffRate(MM->getCMotorGhost()->getOffReaction()->getBareRate());
538  MM->getCMotorGhost()->getOffReaction()->setRate(MM->getCMotorGhost()->getOffReaction()->getBareRate());
539  MM->updateReactionRates();
540  MM->getCMotorGhost()->getOffReaction()->updatePropensity();
541  }
542  int dummy=0;
543  for (auto BB: BranchingPoint::getBranchingPoints()) {
544  dummy++;
545  BB->getCBranchingPoint()->setOffRate(BB->getCBranchingPoint()->getOffReaction()->getBareRate());
546  BB->getCBranchingPoint()->getOffReaction()->setRate(BB->getCBranchingPoint()->getOffReaction()->getBareRate());
547  BB->getCBranchingPoint()->getOffReaction()->updatePropensity();
548  }
549 //STEP 7: Get cylinders, activate filament reactions.
550  for(auto C : _subSystem->getCompartmentGrid()->getCompartments()) {
551  for(auto x : C->getCylinders()) {
552  x->getCCylinder()->activatefilreactions();
553  }}
554  cout<<"Unbinding rates of bound species restored. filament reactions activated"<<endl;
555 //@
556 #ifdef CHEMISTRY
557  _subSystem->updateBindingManagers();
558 #endif
559 #ifdef DYNAMICRATES
560  updateReactionRates();
561 #endif
562 // for(auto o: _outputs) o->print(_numChemSteps);
563  cout<< "Restart procedures completed. Starting original Medyan framework"<<endl;
564  cout << "---" << endl;
565  resetglobaltime();
566  _cController->restart();
567  cout << "Current simulation time = "<< tau() << endl;
568  //restart phase ends
569  }
570 #ifdef CHEMISTRY
571  tauLastSnapshot = tau();
572  oldTau = 0;
573 #endif
574  for(auto o: _outputs) o->print(0);
575 
576  cout << "Starting simulation..." << endl;
577 
578  int i = 1;
579 
580  //if runtime was specified, use this
581  if(!areEqual(_runTime, 0.0)) {
582 
583 #ifdef CHEMISTRY
584  while(tau() <= _runTime) {
585  //run ccontroller
586  if(!_cController->run(_minimizationTime)) {
587  for(auto o: _outputs) o->print(i);
588  break;
589  }
590 
591  //add the last step
592  tauLastSnapshot += tau() - oldTau;
593  tauLastMinimization += tau() - oldTau;
594  tauLastNeighborList += tau() - oldTau;
595 #endif
596 #if defined(MECHANICS) && defined(CHEMISTRY)
597  //run mcontroller, update system
598  if(tauLastMinimization >= _minimizationTime) {
599  _mController->run();
600  updatePositions();
601 
602  tauLastMinimization = 0.0;
603  }
604 
605  if(tauLastSnapshot >= _snapshotTime) {
606  cout << "Current simulation time = "<< tau() << endl;
607  for(auto o: _outputs) o->print(i);
608  i++;
609  tauLastSnapshot = 0.0;
610  }
611 #elif defined(MECHANICS)
612  for(auto o: _outputs) o->print(i);
613  i++;
614 #endif
615 
616 #ifdef DYNAMICRATES
617  updateReactionRates();
618 #endif
619 
620 #ifdef CHEMISTRY
621  // update neighbor lists
622  if(tauLastNeighborList >= _neighborListTime) {
623  updateNeighborLists();
624  tauLastNeighborList = 0.0;
625  }
626 
627  //move the boundary
628  moveBoundary(tau() - oldTau);
629 
630  //special protocols
631  executeSpecialProtocols();
632 
633  oldTau = tau();
634  }
635 #endif
636  }
637  //if run steps were specified, use this
638  if(_runSteps != 0) {
639 
640 #ifdef CHEMISTRY
641  while(totalSteps <= _runSteps) {
642  //run ccontroller
643  if(!_cController->runSteps(_minimizationSteps)) {
644  for(auto o: _outputs) o->print(i);
645  break;
646  }
647 
648  //add the last step
649  stepsLastSnapshot += _minimizationSteps;
650  stepsLastMinimization += _minimizationSteps;
651  stepsLastNeighborList += _minimizationSteps;
652 
653  totalSteps += _minimizationSteps;
654 #endif
655 #if defined(MECHANICS) && defined(CHEMISTRY)
656  //run mcontroller, update system
657  if(stepsLastMinimization >= _minimizationSteps) {
658  _mController->run();
659  updatePositions();
660 
661  stepsLastMinimization = 0;
662  }
663 
664  if(stepsLastSnapshot >= _snapshotSteps) {
665  cout << "Current simulation time = "<< tau() << endl;
666  for(auto o: _outputs) o->print(i);
667  i++;
668  stepsLastSnapshot = 0;
669  }
670 #elif defined(MECHANICS)
671  for(auto o: _outputs) o->print(i);
672  i++;
673 #endif
674 
675 #ifdef DYNAMICRATES
676  updateReactionRates();
677 #endif
678 
679 #ifdef CHEMISTRY
680  // update neighbor lists
681  if(stepsLastNeighborList >= _neighborListSteps) {
682  updateNeighborLists();
683  stepsLastNeighborList = 0;
684  }
685 
686  //move the boundary
687  moveBoundary(tau() - oldTau);
688 
689  //special protocols
690  executeSpecialProtocols();
691  }
692 #endif
693  }
694 
695  //print last snapshots
696  for(auto o: _outputs) o->print(i);
697 
698  chk2 = chrono::high_resolution_clock::now();
699  chrono::duration<double> elapsed_run(chk2-chk1);
700 
701  cout << "Time elapsed for run: dt=" << elapsed_run.count() << endl;
702  cout << "Total simulation time: dt=" << tau() << endl;
703  cout << "Done with simulation!" << endl;
704 }
