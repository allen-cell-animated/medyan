123  if(FSetup.inputFile != "") {

125  filamentData = fp.readFilaments();

127  else {
128  //Create random distribution of filaments
129  default_random_engine generator;
130  uniform_real_distribution<double> dU(0.0,1.0);
131  uniform_real_distribution<double> dUNeg(-1,1);
132 
133  int filamentCounter = 0;
134  while (filamentCounter < FSetup.numFilaments) {
135 
136  double firstX = dU(generator) * SystemParameters::Geometry().compartmentSizeX *
137  SystemParameters::Geometry().NX;
138  double firstY = dU(generator) * SystemParameters::Geometry().compartmentSizeY *
139  SystemParameters::Geometry().NY;
140  double firstZ = dU(generator) * SystemParameters::Geometry().compartmentSizeZ *
141  SystemParameters::Geometry().NZ;
142 
143  double directionX = dUNeg(generator);
144  double directionY = dUNeg(generator);
145  double directionZ = dUNeg(generator);
146 
147  //Create a random filament vector one cylinder long
148  vector<double> firstPoint = {firstX, firstY, firstZ};
149 
150  auto normFactor = sqrt(directionX * directionX +
151  directionY * directionY +
152  directionZ * directionZ);
153 
154  vector<double> direction = {directionX/normFactor,
155  directionY/normFactor,
156  directionZ/normFactor};
157 
158  vector<double> secondPoint = NextPointProjection(firstPoint,
159  (double)SystemParameters::Geometry().cylinderSize - 0.01, direction);
160 
161  if(_subSystem->getBoundary()->within(firstPoint)
162  && _subSystem->getBoundary()->within(secondPoint)) {
163  filamentData.push_back({firstPoint, secondPoint});
164  filamentCounter++;
165  }
166  }
167  }
168  //add filaments
169  _subSystem->addNewFilaments(filamentData);
170  cout << "Done. " << filamentData.size() << " filaments created." << endl;
171 
172  //First update of system
173  updateSystem();
