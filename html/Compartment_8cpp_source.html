45 vector<ReactionBase*> Compartment::generateDiffusionReactions(Compartment* C)
46 {
47  vector<ReactionBase*> rxns;
48 
49  for(auto &sp_this : _species.species()) {
50  int molecule = sp_this->getMolecule();
51  float diff_rate = _diffusion_rates[molecule];
52  if(diff_rate<0) continue;
53 
54  if(C->isActivated()) {
55  Species *sp_neighbour = C->_species.findSpeciesByMolecule(molecule);
56  ReactionBase *R = new DiffusionReaction({sp_this.get(),sp_neighbour},diff_rate);
57  this->addDiffusionReaction(R);
58  rxns.push_back(R);
59  }
60  }
61  return vector<ReactionBase*>(rxns.begin(), rxns.end());
62 }

116 void Compartment::transferSpecies() {
117 
118  //get active neighbors
119  vector<Compartment*> activeNeighbors;
120 
121  for(auto &neighbor : _neighbours)
122  if(neighbor->isActivated())
123  activeNeighbors.push_back(neighbor);
124 
125  assert(activeNeighbors.size() != 0
126  && "Cannot transfer species to another compartment... no neighbors are active");
127 
128  //go through species
129  Species* sp_neighbor;
130  vector<Species*> sp_neighbors;
131 
132  for(auto &sp : _species.species()) {
133 
134  int copyNumber = sp->getN();
135  auto nit = activeNeighbors.begin();
136 
137  while(copyNumber > 0) {
138  sp->down();
139 
140  //choose a random active neighbor
141  auto neighbor = *nit;
142  sp_neighbor = neighbor->findSpeciesByName(sp->getName());
143 
144  //add to list if not already
145  auto spit = find(sp_neighbors.begin(),
146  sp_neighbors.end(),
147  sp_neighbor);
148 
149  if(spit == sp_neighbors.end())
150  sp_neighbors.push_back(sp_neighbor);
151 
152  //increase copy number
153  sp_neighbor->up();
154 
155  //reset if we've looped through
156  if(++nit == activeNeighbors.end())
157  nit = activeNeighbors.begin();
158  copyNumber--;
159  }
160 
161  //activate all reactions changed
162  for(auto spn : sp_neighbors)
163  spn->updateReactantPropensities();
164  }
165 }

195 bool operator==(const Compartment& a, const Compartment& b) {
196  if(a.numberOfSpecies()!=b.numberOfSpecies() or
197  a.numberOfInternalReactions()!=b.numberOfInternalReactions())
198  return false;
199 
200  if(typeid(a)!=typeid(b))
201  return false;
202 
203  bool spec_bool = false;
204  auto sit_pair = mismatch(a._species.species().begin(),
205  a._species.species().end(),
206  b._species.species().begin(),
207  [](const unique_ptr<Species> &A, const unique_ptr<Species> &B)
208  {return (*A)==(*B); });
209  if(sit_pair.first==a._species.species().end())
210  spec_bool=true;
211 
212 
213  bool reac_bool = false;
214  auto rit_pair = mismatch(a._internal_reactions.reactions().begin(),
215  a._internal_reactions.reactions().end(),
216  b._internal_reactions.reactions().begin(),
217  [](const unique_ptr<ReactionBase> &A, const unique_ptr<ReactionBase> &B)
218  {return (*A)==(*B);});
219  if(rit_pair.first==a._internal_reactions.reactions().end())
220  reac_bool=true;
221 
222  return spec_bool && reac_bool;
223 }
