47 vector<ReactionBase*> Compartment::generateDiffusionReactions(Compartment* C)
48 {
49  vector<ReactionBase*> rxns;
50 
51  for(auto &sp_this : _species.species()) {
52  int molecule = sp_this->getMolecule();
53  int diff_rate = _diffusion_rates[molecule];
54  if(diff_rate<0) // Based on a convention that diffusing reactions require positive rates
55  continue;
56 
57  if(C->isActivated()) {
58  Species *sp_neighbour = C->_species.findSpeciesByMolecule(molecule);
59  ReactionBase *R = new Reaction<1,1>({sp_this.get(),sp_neighbour},diff_rate);
60  R->setReactionType(ReactionType::DIFFUSION);
61  this->addDiffusionReactionUnique(unique_ptr<ReactionBase>(R));
62  rxns.push_back(R);
63  }
64  }
65  return vector<ReactionBase*>(rxns.begin(), rxns.end());
66 }
