140 bool ChemNRMImpl::makeStep() {
141 
142  RNodeNRM *rn = _heap.top()._rn;
143  double tau_top = rn->getTau();
144  if(tau_top==numeric_limits<double>::infinity()){
145  cout << "The heap has been exhausted - no more reactions to fire, returning..." << endl;
146  return false;
147  }
148  _t=tau_top;
149  rn->makeStep();
150 #if defined TRACK_ZERO_COPY_N || defined TRACK_UPPER_COPY_N
151  if(!rn->isPassivated()){
152 #endif
153  rn->generateNewRandTau();
154  rn->updateHeap();
155 #if defined TRACK_ZERO_COPY_N || defined TRACK_UPPER_COPY_N
156  }
157 #endif
158  // Updating dependencies
159  ReactionBase *r = rn->getReaction();
160 
161  for(auto rit = r->dependents().begin(); rit!=r->dependents().end(); ++rit){
162  RNodeNRM *rn_other = static_cast<RNodeNRM*>((*rit)->getRnode());
163  double a_old = rn_other->getPropensity();
164  rn_other->reComputePropensity();
165  double tau_new;
166  double tau_old = rn_other->getTau();
167  double a_new = rn_other->getPropensity();
168 #ifdef TRACK_ZERO_COPY_N
169  tau_new = (a_old/a_new)*(tau_old-_t)+_t;
170 #else
171  if(a_new<1.0e-15) // numeric_limits< double >::min()
172  tau_new = numeric_limits<double>::infinity();
173  else if (a_old<1.0e-15){
174  rn_other->generateNewRandTau();
175  tau_new = rn_other->getTau();
176  }
177  else{
178  tau_new = (a_old/a_new)*(tau_old-_t)+_t;
179  }
180 #endif
181  rn_other->setTau(tau_new);
182  rn_other->updateHeap();
183  }
184 
185  // Send signals
186 #ifdef REACTION_SIGNALING
187  r->emitSignal();
188 #endif
189 #ifdef RSPECIES_SIGNALING
190  r->broadcastRSpeciesSignals();
191 #endif
192 
193  syncGlobalTime();
194 
195  return true;
196 }
