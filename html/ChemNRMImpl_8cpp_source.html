140 bool ChemNRMImpl::makeStep() {
141 
142  //try to get a reaction
143  if(_heap.empty()) {
144  cout << "There are no reactions to fire, returning..." << endl;
145  return false;
146  }
147  RNodeNRM *rn = _heap.top()._rn;
148  double tau_top = rn->getTau();
149  if(tau_top==numeric_limits<double>::infinity()){
150  cout << "The heap has been exhausted - no more reactions to fire, returning..." << endl;
151  return false;
152  }
154  //assert heap ordering
155  if(tau_top < _t) {
156  cout << "ERROR: The heap is not correctly sorted, returning..." << endl;
157  cout << "Tau top = " << tau_top << endl;
158  cout << "Tau current = " << _t << endl;
159  cout << "Reaction type = " << rn->getReaction()->getReactionType() << endl;
160  rn->printSelf();
161  return false;
162  }
163 
164  double t_prev = _t;
165 
166  _t=tau_top;
167  syncGlobalTime();
168 
169  rn->makeStep();
170 #if defined TRACK_ZERO_COPY_N || defined TRACK_UPPER_COPY_N
171  if(!rn->isPassivated()){
172 #endif
173  rn->generateNewRandTau();
174  rn->updateHeap();
175 #if defined TRACK_ZERO_COPY_N || defined TRACK_UPPER_COPY_N
176  }
177 #endif
178  // Updating dependencies
179  ReactionBase *r = rn->getReaction();
180 
181  if(r->updateDependencies()) {
182 
183  for(auto rit = r->dependents().begin(); rit!=r->dependents().end(); ++rit){
184 
185  RNodeNRM *rn_other = (RNodeNRM*)((*rit)->getRnode());
186  double a_old = rn_other->getPropensity();
187 
188  //recompute propensity
189  rn_other->reComputePropensity();
190 
191  double tau_new;
192  double tau_old = rn_other->getTau();
193 
194  double a_new = rn_other->getPropensity();
195 
196 #ifdef TRACK_ZERO_COPY_N
197  //recompute tau
198  tau_new = (a_old/a_new)*(tau_old-_t)+_t;
199 #else
200  //recompute tau
201  if(a_new<1.0e-15) // numeric_limits< double >::min()
202  tau_new = numeric_limits<double>::infinity();
203  else if (a_old<1.0e-15){
204  rn_other->generateNewRandTau();
205  tau_new = rn_other->getTau();
206  }
207  else{
208  tau_new = (a_old/a_new)*(tau_old-_t)+_t;
209  }
210 #endif
211  if(boost::math::isnan(tau_new)){tau_new=numeric_limits<double>::infinity();}
213  if(tau_new < _t) {
214 
215  cout << "WARNING: Generated tau may be incorrect. " << endl;
216 
217  cout << "Tau new = " << tau_new << endl;
218  cout << "Tau old = " << tau_old << endl;
219  cout << "Current global t = " << _t << endl;
220  cout << "Previous global t = " << t_prev << endl;
221  cout << "a_old = " << a_old << endl;
222  cout << "a_new = " << a_new << endl;
223 
224  cout << "Reaction type = " << rn->getReaction()->getReactionType() << endl;
225 
226 
227  rn->printSelf();
228  rn_other->printSelf();
229  }
230 
231  rn_other->setTau(tau_new);
232  rn_other->updateHeap();
233  }
234  }
235 #ifdef REACTION_SIGNALING
236  // Send signal
237  r->emitSignal();
238 #endif
239  return true;
240 }
