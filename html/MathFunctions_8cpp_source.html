76  tuple<vector<double>, vector<double>> branchProjection(const vector<double>& n,
77  const vector<double>& p,
78  double l, double m, double theta){
79  //get random permutation from p
80  vector<double> r = {p[0] + Rand::randDouble(-1, 1),
81  p[1] + Rand::randDouble(-1, 1),
82  p[2] + Rand::randDouble(-1, 1)};
83 
84  //construct vector z which is r-p
85  auto z = twoPointDirection(p, r);
86 
87  //construct u and v, which creates an orthogonal set n, u, v
88  auto u = crossProduct(n, z);
89  auto v = crossProduct(n, u);
90 
91  normalize(u); normalize(v);
92 
93  //find random point on circle defining the branching point
94  double thetaRandom = Rand::randDouble(0, 2*M_PI);
95  vector<double> bp1;
96  bp1.push_back(p[0] + l * (u[0] * cos(thetaRandom) + v[0] * sin(thetaRandom)));
97  bp1.push_back(p[1] + l * (u[1] * cos(thetaRandom) + v[1] * sin(thetaRandom)));
98  bp1.push_back(p[2] + l * (u[2] * cos(thetaRandom) + v[2] * sin(thetaRandom)));
99 
100  //now find the second point
101  vector<double> newP;
102  double dist = m * cos(theta);
103  newP.push_back(p[0] + n[0] * dist);
104  newP.push_back(p[1] + n[1] * dist);
105  newP.push_back(p[2] + n[2] * dist);
106  double newL = (l + m * sin(theta));
107 
108  vector<double> bp2;
109  bp2.push_back(newP[0] + newL * (u[0] * cos(thetaRandom) + v[0] * sin(thetaRandom)));
110  bp2.push_back(newP[1] + newL * (u[1] * cos(thetaRandom) + v[1] * sin(thetaRandom)));
111  bp2.push_back(newP[2] + newL * (u[2] * cos(thetaRandom) + v[2] * sin(thetaRandom)));
112 
113  //get direction
114  auto direction = twoPointDirection(bp1, bp2);
115 
116  return tuple<vector<double>, vector<double>>(direction, bp1);
117  }
