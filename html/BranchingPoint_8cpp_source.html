72 BranchingPoint::~BranchingPoint() noexcept {
73 
74 #ifdef MECHANICS
75  //offset the branching cylinder's bead by a little for safety
76  auto msize = SysParams::Geometry().monomerSize[_c1->getType()];
77 
78  vector<double> offsetCoord =
79  {(Rand::randInteger(0,1) ? -1 : +1) * Rand::randDouble(msize, 2 * msize),
80  (Rand::randInteger(0,1) ? -1 : +1) * Rand::randDouble(msize, 2 * msize),
81  (Rand::randInteger(0,1) ? -1 : +1) * Rand::randDouble(msize, 2 * msize)};
82 
83  auto b = _c2->getFirstBead();
84 
85  b->coordinate[0] += offsetCoord[0];
86  b->coordinate[1] += offsetCoord[1];
87  b->coordinate[2] += offsetCoord[2];
88 #endif
89 
90 
91 #ifdef CHEMISTRY
92  //mark the correct species on the minus end of the branched
93  //filament. If this is a filament species, change it to its
94  //corresponding minus end. If a plus end, release a diffusing
95  //or bulk species, depending on the initial reaction.
96  CMonomer* m = _c2->getCCylinder()->getCMonomer(0);
97  short speciesFilament = m->activeSpeciesFilament();
98 
99  //there is a filament species, mark its corresponding minus end
100  if(speciesFilament != -1) {
101  m->speciesMinusEnd(speciesFilament)->up();
102 
103  //unmark the filament and bound species
104  m->speciesFilament(speciesFilament)->down();
105  m->speciesBound(SysParams::Chemistry().brancherBoundIndex[_c1->getType()])->down();
106  }
107  //mark the free species instead
108  else {
109  //find the free species
110  Species* speciesFilament = m->speciesFilament(m->activeSpeciesPlusEnd());
111 
112  string speciesName = SpeciesNamesDB::removeUniqueFilName(speciesFilament->getName());
113  string speciesFirstChar = speciesName.substr(0,1);
114 
115  //find the free monomer, either bulk or diffusing
116  Species* freeMonomer = nullptr;
117  auto grid = _subSystem->getCompartmentGrid();
118 
119  Species* dMonomer = _compartment->findSpeciesByName(speciesName);
120  Species* dfMonomer = _compartment->findSpeciesByName(speciesFirstChar);
121 
122  Species* bMonomer = grid->findSpeciesBulkByName(speciesName);
123  Species* bfMonomer = grid->findSpeciesBulkByName(speciesFirstChar);
124 
125  //try diffusing
126  if(dMonomer != nullptr) freeMonomer = dMonomer;
127  // try bulk
128  else if(bMonomer != nullptr) freeMonomer = bMonomer;
129  //diffusing, remove all but first char
130  else if(dfMonomer != nullptr) freeMonomer = dfMonomer;
131  //bulk, remove all but first char
132  else if(bfMonomer != nullptr) freeMonomer = bfMonomer;
133  //could not find. exit ungracefully
134  else {
135  cout << "In unbranching reaction, could not find corresponding " <<
136  "diffusing species of filament species " << speciesName <<
137  ". Exiting." << endl;
138  exit(EXIT_FAILURE);
139  }
140 
141  //remove the filament from the system
142  Filament *bf = (Filament*)(_c2->getParent());
143  _subSystem->removeTrackable<Filament>(bf);
144 
145  delete bf;
146 
147  //mark species, update reactions
148  freeMonomer->up();
149  freeMonomer->updateReactantPropensities();
150  }
151 #endif
152  //reset branching cylinder
153  _c1->setBranchingCylinder(nullptr);
154 }
