58 TEST(ChemNRMTest, SimpleSteadyState) {
59  int Nstart = 16;
60  SpeciesBulk A1("A1", Nstart);
61  SpeciesBulk A2("A2", 0);
62  // A1 <-> A2 with the same forward and backward rates; [A]~[B] at steady state
63  Reaction<1,1> r1 = { {&A1,&A2}, 100.0 };
64  Reaction<1,1> r2 = { {&A2,&A1}, 100.0 };
65 
66  ChemSim::setInstance(new ChemNRMImpl());
67 
68  ChemSim::addReaction(&r1);
69  ChemSim::addReaction(&r2);
70 
71  ChemSim::initialize();
72 
73  ChemSim::run(1000);
74 
75  accumulator_set<int, stats<tag::variance(immediate)>> accA1;
76  accumulator_set<int, stats<tag::mean>> accA2;
77  accumulator_set<int, stats<tag::covariance<double, tag::covariate1> > > accCov;
78  int N_SAMPLE_POINTS=1000;
79  for(int i=0;i<N_SAMPLE_POINTS;++i){
80  ChemSim::run(100);
81  accA1(A1.getN());
82  accCov(A1.getN(), covariate1 = A2.getN());
83  }
84  double A1mean = mean(accA1);
85  double A1var = variance(accA1);
86  double var_expected = double(Nstart)/2/2;
87  double mean_error = sqrt(var_expected/N_SAMPLE_POINTS);
88  EXPECT_TRUE(fabs(A1mean-Nstart/2)<10*mean_error);
89  EXPECT_TRUE(fabs(var_expected-A1var)<0.1*var_expected);// within 10% of the expected variance
90  EXPECT_FLOAT_EQ(-1.0,covariance(accCov)/(A1var));
91 
92 }
