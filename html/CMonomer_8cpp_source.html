32 CMonomer::CMonomer(const CMonomer& rhs, Compartment* c)
33 
34  : CMonomer(rhs._filamentType) {
35 
36  for(int i = 0; i < _numFSpecies[_filamentType]; i++) {
37 
38  //clone and add to array
39  SpeciesFilament* s = rhs._speciesFilament[i];
40  SpeciesFilament* sNew = s->clone();
41 
42  c->addSpeciesUnique(unique_ptr<Species>(sNew));
43  _speciesFilament[i] = sNew;
44  }
45 
46  //For bound species, transfer the CBound (if any)
47 
48  for(int i = 0; i < _numBSpecies[_filamentType]; i++) {
49 
50  //clone and add to array
51  SpeciesBound* s = rhs._speciesBound[i];
52  SpeciesBound* sNew = s->clone();
53 
54  c->addSpeciesUnique(unique_ptr<Species>(sNew));
55  _speciesBound[i] = sNew;
56 
57  //update cbound
58  CBound* cBound = s->getCBound();
59  if(cBound != nullptr) {
60  //set species
61  if(cBound->getFirstSpecies() == s)
62  cBound->setFirstSpecies(sNew);
63  else
64  cBound->setSecondSpecies(sNew);
65  }
66  }
67 }

185 bool CMonomer::isConsistent() {
186 
187  //check all species between 0 and 1 inclusive
188  for(int i = 0; i < _numFSpecies[_filamentType]; i++) {
189 
190  if(!areEqual(_speciesFilament[i]->getN(), 1.0) &&
191  !areEqual(_speciesFilament[i]->getN(), 0.0)) {
192 
193  cout << _speciesFilament[i]->getName() << " has an invalid copy number. It is = "
194  << _speciesFilament[i]->getN() << " and is at species index " << i << "." << endl;
195 
196  return false;
197  }
198  }
199  //check filament species
200  if(activeSpeciesFilament() != -1 &&
201  (activeSpeciesPlusEnd() != -1 ||
202  activeSpeciesMinusEnd() != -1)) {
203 
204  cout << "Has a simultaneously active filament and plus/minus end species." << endl;
205 
206  return false;
207  }
208 
209  return true;
210 }
