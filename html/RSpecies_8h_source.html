447  void computeAverageN() {
448 
449  double totalTau = tau() - _firstTau;
450  double average = 0;
451 
452  //average values
453  for(auto it : _copyNumbers)
454  average += it.first * it.second;
455 
456  average /= totalTau;
457 
458  //clear map and return
459  _copyNumbers.clear();
460 
461  _average = average;
462  }

464 public:
469  RSpeciesAvg (Species &parent, species_copy_t n=0, species_copy_t ulim=max_ulim)
470 
471  : RSpecies(parent, n, ulim) {
472 
473  _type = RSpeciesType::AVG;
474 
475  //set first average to n, first time update
476  _average = n;
477  _localTau = _firstTau = tau();
478  }

511  virtual void up() {
512 
513  // if initialization, set avg to be true
514  // copy number and return
515  double deltaTau = tau() - _localTau;
516 
517  if(areEqual(deltaTau, 0.0)) {
518 
519  _average = ++_n;
520  return;
521  }
522 
523  //add old n to map
524  _copyNumbers[_n] += deltaTau;
525 
526  //compute a new avg if we need it
527  if(++_eventCount > _numEvents) {
528  computeAverageN();
529 
530  _eventCount = 0;
531  _firstTau = tau();
532 
533  _newAvg = true;
534  }
535  else _newAvg = false;
536 
537  //increase copy number, reset local tau
538  _n++; _localTau = tau();
539 
540 #ifdef TRACK_ZERO_COPY_N
541  if(_n==1)
542  activateAssocReactantReactions();
543 #endif
544 #ifdef TRACK_UPPER_COPY_N
545  if(_n==_ulim)
546  passivateAssocProductReactions();
547 #endif
548 
549 #ifdef RSPECIES_SIGNALING
550  if(isSignaling()) emitSignal(+1);
551 #endif
552  }
553

568  //add old n to map

570 
571  //compute a new avg if we need it
572  if(++_eventCount > _numEvents) {

574

580 
581  _newAvg = true;
