275 vector<vector<double> > Filament::straightFilamentProjection(vector<vector<double>>& v, int numBeads){
276 
277  vector<vector<double>> coordinate;
278  vector<double> tmpVec (3, 0);
279  vector<double> tau (3, 0);
280  double invD = 1/TwoPointDistance(v[1], v[0]);
281  tau[0] = invD * ( v[1][0] - v[0][0] );
282  tau[1] = invD * ( v[1][1] - v[0][1] );
283  tau[2] = invD * ( v[1][2] - v[0][2] );
284 
285  for (int i = 0; i<numBeads; i++) {
286 
287  tmpVec[0] = v[0][0] + SystemParameters::Geometry().cylinderSize * i * tau[0];
288  tmpVec[1] = v[0][1] + SystemParameters::Geometry().cylinderSize * i * tau[1];
289  tmpVec[2] = v[0][2] + SystemParameters::Geometry().cylinderSize * i * tau[2];
290 
291  coordinate.push_back(tmpVec);
292  }
293  return coordinate;
294 }

296 vector<vector<double> > Filament::zigZagFilamentProjection(vector<vector<double>>& v, int numBeads){
297 
298  vector<vector<double>> coordinate;
299  vector<double> tmpVec (3, 0);
300  vector<double> tau (3, 0);
301  double invD = 1/TwoPointDistance(v[1], v[0]);
302  tau[0] = invD * ( v[1][0] - v[0][0] );
303  tau[1] = invD * ( v[1][1] - v[0][1] );
304  tau[2] = invD * ( v[1][2] - v[0][2] );
305 
306  vector<double> perptau = {-tau[1], tau[0], tau[2]};
307 
308 
309  for (int i = 0; i<numBeads; i++) {
310 
311  if(i%2 == 0) {
312  tmpVec[0] = v[0][0] + SystemParameters::Geometry().cylinderSize * i * tau[0];
313  tmpVec[1] = v[0][1] + SystemParameters::Geometry().cylinderSize * i * tau[1];
314  tmpVec[2] = v[0][2] + SystemParameters::Geometry().cylinderSize * i * tau[2];
315  }
316  else {
317  tmpVec[0] = v[0][0] + SystemParameters::Geometry().cylinderSize * i * perptau[0];
318  tmpVec[1] = v[0][1] + SystemParameters::Geometry().cylinderSize * i * perptau[1];
319  tmpVec[2] = v[0][2] + SystemParameters::Geometry().cylinderSize * i * perptau[2];
320  }
321 
322  coordinate.push_back(tmpVec);
323  }
324  return coordinate;
325 }
