72 Filament::Filament(SubSystem* s, short filamentType, vector<vector<double> >& position,
73  int numBeads, string projectionType)
74 
75  : Trackable(), _subSystem(s), _filType(filamentType), _ID(_filaments.getID()) {
76 
77  //create a projection of beads
78  vector<vector<double>> tmpBeadsCoord;
79 
80  //straight projection
81  if(projectionType == "STRAIGHT")
82  tmpBeadsCoord = straightFilamentProjection(position, numBeads);
83  //zigzag projection
84  else if(projectionType == "ZIGZAG")
85  tmpBeadsCoord = zigZagFilamentProjection(position, numBeads);
86  //arc projection
87  else if(projectionType == "ARC")
88  tmpBeadsCoord = arcFilamentProjection(position, numBeads);
89  //predefined projection aravind sep 9, 15
90  else if(projectionType == "PREDEFINED")
91  tmpBeadsCoord = predefinedFilamentProjection(position, numBeads);
92 
93  //create beads
94  auto direction = twoPointDirection(tmpBeadsCoord[0], tmpBeadsCoord[1]);
95 
96  Bead* b1 = _subSystem->addTrackable<Bead>(tmpBeadsCoord[0], this, 0);
97  Bead* b2 = _subSystem->addTrackable<Bead>(tmpBeadsCoord[1], this, 1);
98 
99  Cylinder* c0 = _subSystem->addTrackable<Cylinder>(this, b1, b2, _filType, 0,
100  false, false, true);
101 
102  c0->setPlusEnd(true);
103  c0->setMinusEnd(true);
104  _cylinderVector.push_back(c0);
105 
106  for (int i = 2; i<numBeads; i++)
107  extendPlusEnd(tmpBeadsCoord[i]);
108 
109  //set plus end marker
110  _plusEndPosition = numBeads - 1;
111 }

478 Filament* Filament::sever(int cylinderPosition) {
479 
480  int vectorPosition = 0;
481 
482  //loop through cylinder vector, find position
483  for(auto &c : _cylinderVector) {
484 
485  if(c->getPosition() == cylinderPosition) break;
486  else vectorPosition++;
487  }
488 
489  //if vector position is zero, we can't sever. return null
490  if(vectorPosition == 0) return nullptr;
491 
492 #ifdef CHEMISTRY
493  //if any of the cylinders are only one monomer long, we can't sever
494  CCylinder* ccBack = _cylinderVector[vectorPosition - 1]->getCCylinder();
495  CMonomer* cm = ccBack->getCMonomer(ccBack->getSize() - 1);
496 
497  if(cm->activeSpeciesMinusEnd() != -1) return nullptr;
498 #endif
499 
500  //create a new filament
501  Filament* newFilament = _subSystem->addTrackable<Filament>(_subSystem, _filType);
502 
503  //Split the cylinder vector at position, transfer cylinders to new filament
504  for(int i = vectorPosition; i > 0; i--) {
505 
506  Cylinder* c = _cylinderVector.front();
507  _cylinderVector.pop_front();
508 
509  newFilament->addChild(unique_ptr<Component>(c));
510  newFilament->_cylinderVector.push_back(c);
511  }
512  //new front of new filament, back of old
513  auto c1 = newFilament->_cylinderVector.back();
514  auto c2 = _cylinderVector.front();
515 
517  Bead* newB = _subSystem->addTrackable<Bead>(*(c2->getFirstBead()));
518  Bead* oldB = c2->getFirstBead();
519 
520  //offset these beads by a little for safety
521  auto msize = SysParams::Geometry().monomerSize[_filType];
522 
523  vector<double> offsetCoord =
524  {(Rand::randInteger(0,1) ? -1 : +1) * Rand::randDouble(msize, 2 * msize),
525  (Rand::randInteger(0,1) ? -1 : +1) * Rand::randDouble(msize, 2 * msize),
526  (Rand::randInteger(0,1) ? -1 : +1) * Rand::randDouble(msize, 2 * msize)};
527 
528  oldB->coordinate[0] += offsetCoord[0];
529  oldB->coordinate[1] += offsetCoord[1];
530  oldB->coordinate[2] += offsetCoord[2];
531 
532  newB->coordinate[0] += -offsetCoord[0];
533  newB->coordinate[1] += -offsetCoord[1];
534  newB->coordinate[2] += -offsetCoord[2];
535 
536  c1->setSecondBead(newB);
537 
538  //set plus and minus ends
539  c1->setPlusEnd(true);
540  c2->setMinusEnd(true);
541 
542 #ifdef CHEMISTRY
543  //mark the plus and minus ends of the new and old filament
544  CCylinder* cc1 = c1->getCCylinder();
545  CCylinder* cc2 = c2->getCCylinder();
546 
547  CMonomer* m1 = cc1->getCMonomer(cc1->getSize() - 1);
548  CMonomer* m2 = cc2->getCMonomer(0);
549 
550  short filamentInt1 = m1->activeSpeciesFilament();
551  short filamentInt2 = m2->activeSpeciesFilament();
552 
553  //plus end
554  m1->speciesFilament(filamentInt1)->down();
555  m1->speciesPlusEnd(filamentInt1)->up();
556 
557  for(auto j : SysParams::Chemistry().bindingIndices[_filType])
558  m1->speciesBound(j)->down();
559 
560  //minus end
561  m2->speciesFilament(filamentInt2)->down();
562  m2->speciesMinusEnd(filamentInt2)->up();
563 
564  for(auto j : SysParams::Chemistry().bindingIndices[_filType])
565  m2->speciesBound(j)->down();
566 
567  //remove any cross-cylinder rxns between these two cylinders
568  cc1->removeCrossCylinderReactions(cc2);
569  cc2->removeCrossCylinderReactions(cc1);
570 #endif
571 
572  return newFilament;
573 }

575 vector<vector<double>> Filament::straightFilamentProjection(vector<vector<double>>& v, int numBeads) {
576 
577  vector<vector<double>> coordinate;
578  vector<double> tmpVec (3, 0);
579  vector<double> tau (3, 0);
580  double invD = 1/twoPointDistance(v[1], v[0]);
581  tau[0] = invD * ( v[1][0] - v[0][0] );
582  tau[1] = invD * ( v[1][1] - v[0][1] );
583  tau[2] = invD * ( v[1][2] - v[0][2] );
584 
585  for (int i = 0; i<numBeads; i++) {
586 
587  tmpVec[0] = v[0][0] + SysParams::Geometry().cylinderSize[_filType] * i * tau[0];
588  tmpVec[1] = v[0][1] + SysParams::Geometry().cylinderSize[_filType] * i * tau[1];
589  tmpVec[2] = v[0][2] + SysParams::Geometry().cylinderSize[_filType] * i * tau[2];
590 
591  coordinate.push_back(tmpVec);
592  }
593  return coordinate;
594 }

596 vector<vector<double>> Filament::zigZagFilamentProjection(vector<vector<double>>& v, int numBeads){
597 
598  vector<vector<double>> coordinate;
599  vector<double> tmpVec (3, 0);
600  vector<double> tau (3, 0);
601  double invD = 1/twoPointDistance(v[1], v[0]);
602  tau[0] = invD * ( v[1][0] - v[0][0] );
603  tau[1] = invD * ( v[1][1] - v[0][1] );
604  tau[2] = invD * ( v[1][2] - v[0][2] );
605 
606  vector<double> perptau = {-tau[1], tau[0], tau[2]};
607 
608 
609  for (int i = 0; i<numBeads; i++) {
610 
611  if(i%2 == 0) {
612  tmpVec[0] = v[0][0] + SysParams::Geometry().cylinderSize[_filType] * i * tau[0];
613  tmpVec[1] = v[0][1] + SysParams::Geometry().cylinderSize[_filType] * i * tau[1];
614  tmpVec[2] = v[0][2] + SysParams::Geometry().cylinderSize[_filType] * i * tau[2];
615  }
616  else {
617  tmpVec[0] = v[0][0] + SysParams::Geometry().cylinderSize[_filType] * i * perptau[0];
618  tmpVec[1] = v[0][1] + SysParams::Geometry().cylinderSize[_filType] * i * perptau[1];
619  tmpVec[2] = v[0][2] + SysParams::Geometry().cylinderSize[_filType] * i * perptau[2];
620  }
621 
622  coordinate.push_back(tmpVec);
623  }
624  return coordinate;
625 }

629 void marsagila(vector<double>&v) {
630 
631  double d1,d2,d3;
632  double *x=new double[3];
633  d1=2*Rand::randDouble(0,1)-1;
634  d2=2*Rand::randDouble(0,1)-1;
635  d3=pow(d1,2)+pow(d2,2);
636 
637  while(d3>=1) {
638  d1=2*Rand::randDouble(0,1)-1;
639  d2=2*Rand::randDouble(0,1)-1;
640  d3=pow(d1,2)+pow(d2,2);
641  }
642 
643  x[0]=2.0*d1*pow((1.0-d3),0.5);
644  x[1]=2.0*d2*pow(1.0-d3,0.5);
645  x[2]=1.0-2.0*d3;
646  v[0]=2.0*d1*pow((1.0-d3),0.5);
647  v[1]=2.0*d2*pow(1.0-d3,0.5);
648  v[2]=1.0-2.0*d3;
649 }
