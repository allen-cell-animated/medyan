91 bool ChemGillespieImpl::makeStep() {
92  RNodeGillespie *rn_selected = nullptr;
93 
94  //Gillespie algorithm's first step; We assume that _a_total is up to date
95  double tau = generateTau(_a_total);
96  _t+=tau;
97  syncGlobalTime();
98 
99  //Gillespie algorithm's second step: finding which reaction happened;
100  double mu = _a_total*generateUniform();
101  double rates_sum = 0;
102  for (auto &x : _map_rnodes){
103 
104  auto rn = x.second.get();
105  if(rn->getReaction()->isPassivated()) continue;
106 
107  rates_sum+=rn->getPropensity();
108 
109  if(rates_sum>mu){
110  rn_selected = rn;
111  break;
112  }
113  }
114  if(rn_selected==nullptr){
115  cout << "ChemGillespieImpl::makeStep() for loop: rates_sum=" <<
116  rates_sum << ", mu=" << mu << ", _a_total=" << _a_total << endl;
117  throw runtime_error("ChemGillespieImpl::makeStep(): No Reaction was selected during the Gillespie step!");
118  }
119  double a_new, a_penult;
120  rn_selected->makeStep();
121  if(!rn_selected->isPassivated()){
122  rn_selected->reComputePropensity();
123  a_new = rn_selected->getPropensity();
124  a_penult = rn_selected->getPenultStepPropensity();
125  _a_total = _a_total - a_penult + a_new;
126  }
127 
128  // Updating dependencies
129  ReactionBase *r = rn_selected->getReaction();
130 
131  if(r->updateDependencies()) {
132  for(auto rit = r->dependents().begin(); rit!=r->dependents().end(); ++rit){
133  RNodeGillespie *rn_other = (RNodeGillespie*)((*rit)->getRnode());
134  rn_other->reComputePropensity();
135  a_new = rn_other->getPropensity();
136  a_penult = rn_other->getPenultStepPropensity();
137  _a_total = _a_total - a_penult + a_new;
138  }
139  }
140 
141  // Send signal
142 #ifdef REACTION_SIGNALING
143  r->emitSignal();
144 #endif
145  syncGlobalTime();
146  return true;
147 }
