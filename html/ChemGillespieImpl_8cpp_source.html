87 bool ChemGillespieImpl::makeStep() {
88  RNodeGillespie *rn_selected = nullptr;
89 
90  //Gillespie algorithm's first step; We assume that _a_total is up to date
91  double tau = generateTau(_a_total);
92  _t+=tau;
93  syncGlobalTime();
94 
95  //Gillespie algorithm's second step: finding which reaction happened;
96  double mu = _a_total*generateUniform();
97  double rates_sum = 0;
98  for (auto &x : _map_rnodes){
99  auto rn = x.second.get();
100  if(rn->getReaction()->isPassivated())
101  continue;
102  rates_sum+=rn->getPropensity();
103 
104  if(rates_sum>mu){
105  rn_selected = rn;
106  break;
107  }
108  }
109 
110  if(rn_selected==nullptr){
111  cout << "ChemGillespieImpl::makeStep() for loop: rates_sum=" << rates_sum << ", mu="
112  << mu << ", _a_total=" << _a_total << endl;
113  throw runtime_error("ChemGillespieImpl::makeStep(): No Reaction was selected during the Gillespie step!");
114  }
115  double a_new, a_penult;
116  rn_selected->makeStep();
117  if(!rn_selected->isPassivated()){
118  rn_selected->reComputePropensity();
119  a_new = rn_selected->getPropensity();
120  a_penult = rn_selected->getPenultStepPropensity();
121  _a_total = _a_total - a_penult + a_new;
122  }
123 
124  // Updating dependencies
125  ReactionBase *r = rn_selected->getReaction();
126  for(auto rit = r->dependents().begin(); rit!=r->dependents().end(); ++rit){
127  RNodeGillespie *rn_other = static_cast<RNodeGillespie*>((*rit)->getRnode());
128  rn_other->reComputePropensity();
129  a_new = rn_other->getPropensity();
130  a_penult = rn_other->getPenultStepPropensity();
131  _a_total = _a_total - a_penult + a_new;
132  }
133 
134  // Send signals
135 #ifdef REACTION_SIGNALING
136  r->emitSignal();
137 #endif
138 #ifdef RSPECIES_SIGNALING
139  r->broadcastRSpeciesSignals();
140 #endif
141  syncGlobalTime();
142  return true;
143 }
