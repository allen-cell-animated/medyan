29 class PlaneBoundaryElement : public BoundaryElement {
30 
31 friend class BoundaryCubic;
32 
33 private:
35  double _a, _b, _c, _d;
36 
37 public:
39  PlaneBoundaryElement(vector<double> coords,
40  vector<double> normal,
41  double repulsConst,
42  double screenLength)
43 
44  : BoundaryElement(coords, repulsConst, screenLength) {
45 
47  _a = normal[0]; _b = normal[1]; _c = normal[2];
48  _d = -_a * _coords[0] - _b * _coords[1] - _c * _coords[2];
49  }
50 
51  virtual double distance(const vector<double>& point) {
52 
53  return (_a * point[0] + _b * point[1] + _c * point[2] + _d) /
54  sqrt(pow(_a, 2) + pow(_b, 2) + pow(_c, 2));
55  }
56 
57  virtual double stretchedDistance(const vector<double>& point,
58  const vector<double>& force,
59  double d) {
60 
61 
62  vector<double> movedPoint = {point[0] + d*force[0],
63  point[1] + d*force[1],
64  point[2] + d*force[2]};
65  return distance(movedPoint);
66 
67  }
68 
69  virtual const vector<double> normal(const vector<double>& point) {
70 
71  return vector<double>{_a, _b, _c};
72  }
73 
74  virtual void updateCoords(const vector<double> newCoords) {
75 
76  _coords = newCoords;
77 
78  //also update plane params
79  _d = -_a * _coords[0] - _b * _coords[1] - _c * _coords[2];
80  }
81 };

139  CylindricalZBoundaryElement(vector<double> coords,

143  double screenLength)
144 
145  : BoundaryElement(coords, repulsConst, screenLength),
146  _radius(radius), _height(height) {}
147 
148  virtual double distance(const vector<double>& point) {
149 
150 
152  if(point[2] > (_coords[2] + _height / 2) ||
153  point[2] < (_coords[2] - _height / 2))
154 
155  return numeric_limits<double>::infinity();
156 
157  return _radius - twoPointDistance({_coords[0],_coords[1], 0},
158  { point[0], point[1], 0});
159  }
160 
161  virtual double stretchedDistance(const vector<double>& point,
162  const vector<double>& force,
163  double d) {
164 
165  // check z coordinate. If outside, return infinity
166  if((point[2] + d * force[2]) > (_coords[2] + _height / 2) ||
167  (point[2] + d * force[2]) < (_coords[2] - _height / 2))
168 
169  return numeric_limits<double>::infinity();
170 
171  vector<double> movedPoint{point[0] + d * force[0],
172  point[1] + d * force[1],
173  point[2] + d * force[2]};
174 
175  return distance(movedPoint);
176 
177  }
