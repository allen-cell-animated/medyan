42 void CGMethod::makeBracket(ForceFieldManager &FFM, double &ax, double &bx, double &cx, double &fa, double &fb, double &fc) {
43 
44  const double GLIMIT = 100.0, TINY = 1.0e-20;
45  double ulim, u, r, q, fu;
46 
47  fa = FFM.computeEnergy(ax);
48  fb = FFM.computeEnergy(bx);
49 
50  //go in correct direction
51  if (fb > fa){
52  swap(ax, bx);
53  swap(fb, fa);
54  }
55 
56  //first guess
57  cx = bx + PHI * (bx - ax);
58  fc = FFM.computeEnergy(cx);
59 
60  while(fb > fc) {
61 
62  r = (bx - ax) * (fb - fc);
63  q = (bx - cx) * (fb - fa);
64  u = bx - ((bx - cx) * q - (bx - ax) * r) / (2.0 * sign(max(fabs(q - r), TINY), q - r));
65 
66  ulim = bx + GLIMIT * (cx - bx);
67 
68  if ((bx - u) * (u - cx) > 0.0) {
69  fu = FFM.computeEnergy(u);
70  if (fu < fc) {
71  ax = bx;
72  bx = u;
73  fa = fb;
74  fb = fu;
75  return;
76  }
77  else if (fu > fb) {
78  cx = u;
79  fc = fu;
80  return;
81  }
82  u = cx + PHI * (cx - bx);
83  fu = FFM.computeEnergy(u);
84  }
85  else if ((cx - u) * (u - ulim) > 0.0) {
86 
87  fu = FFM.computeEnergy(u);
88  if (fu < fc) {
89  shift3(bx, cx, u, cx + PHI * (cx - bx));
90  shift3(fb, fc, fu, FFM.computeEnergy(u));
91  }
92  }
93  else if((u - ulim) * (ulim - cx) >= 0.0) {
94 
95  u = ulim;
96  fu = FFM.computeEnergy(u);
97  }
98  else {
99  u = cx + PHI * (cx - bx);
100  fu = FFM.computeEnergy(u);
101  }
102  shift3(ax, bx, cx, u);
103  shift3(fa, fb, fc, fu);
104  }
105 }

191  if (FFM.computeEnergy(x1) >= FFM.computeEnergy(x2) ){

263 double CGMethod::backtrackingLineSearch(ForceFieldManager& FFM) {
264 
265  //Forces are used as directions of outer loop minimization (CG). ForceAux -- forces on the beads.
266  double directionDotForce = 0.0;
267  double maxDirection = 0.0;
268 
269  for(auto it: *BeadDB::instance()) {
270  directionDotForce += it->calcDotForceProduct();
271  for(int i=0 ; i < 3; i++) maxDirection = max(maxDirection, fabs(it->force[i]));
272  }
273  //return error if in wrong direction
274  directionDotForce /= BeadDB::instance()->size();
275  if(directionDotForce < 0.0) return -1.0;
276 
277  //return zero if no forces
278  if(maxDirection == 0.0) return 0.0;
279 
280  //calculate first lambda. cannot be greater than lambda max
281  double lambda = min(LAMBDAMAX, MAXDIST / maxDirection);
282  double currentEnergy = FFM.computeEnergy(0.0);
283 
284  //backtracking loop
285  while(true) {
286 
287  double energyLambda = FFM.computeEnergy(lambda);
288  double idealEnergyChange = -BACKTRACKSLOPE * lambda * directionDotForce;
289  double energyChange = energyLambda - currentEnergy;
290 
291  if(energyChange <= idealEnergyChange) {
292  _energyChangeCounter = 0;
293  return lambda;
294  }
295 
296  //reduce lambda
297  lambda *= LAMBDAREDUCE;
298 
299  //cout << "lambda reduced" << endl;
300 
301  if(lambda <= 0 || idealEnergyChange >= -LSENERGYTOL) {
302 
303  if(energyChange < 0) {
304  _energyChangeCounter = 0;
305  return lambda;
306  }
307  else {
308  _energyChangeCounter++;
309  return 0.0;
310  }
311  }
312  }

356

358  _energyChangeCounter = 0;

363  double energyChange = energy - energyPrev;
364

366  _energyChangeCounter = 0;
