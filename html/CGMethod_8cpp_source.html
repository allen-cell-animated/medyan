139 double CGMethod::backtrackingLineSearch(ForceFieldManager& FFM, double MAXDIST,
140  double LAMBDAMAX) {
141 
142  double f = maxF();
143 
144  //return zero if no forces
145  if(f == 0.0) return 0.0;
146 
147  //calculate first lambda
148  double lambda = min(LAMBDAMAX, MAXDIST / f);
149  double currentEnergy = FFM.computeEnergy(0.0);
150 
151  //backtracking loop
152  while(true) {
153 
154  //new energy when moved by lambda
155  double energyLambda = FFM.computeEnergy(lambda);
156 
157  double idealEnergyChange = -BACKTRACKSLOPE * lambda * allFDotFA();
158  double energyChange = energyLambda - currentEnergy;
159 
160  //return if ok
161  if(energyChange <= idealEnergyChange) return lambda;
162 
163  //reduce lambda
164  lambda *= LAMBDAREDUCE;
165 
166  if(lambda <= 0.0 || lambda <= LAMBDATOL)
167  return 0.0;
168  }
169 }
