40

87  }

93  <<endl;
94 
95  return false;

104  return false;

112 
113  return false;
114  }

116  chem.speciesMotor[filType].size() != CParams.numMotorSpecies[filType]) {

396 bool SysParams::checkDyRateParameters(DynamicRateType& dy) {
397 
398  //check types match number of species
399  if(dy.dFPolymerizationType.size() != CParams.numFilaments &&
400  !dy.dFPolymerizationType.empty()) {
401  cout << "Number of filament dynamic rate polymerization forms must" <<
402  " match the number of filaments. Exiting." << endl;
403  return false;
404  }
405 
406  if(dy.dLUnbindingType.size() != sum(CParams.numLinkerSpecies) &&
407  !dy.dLUnbindingType.empty()) {
408  cout << "Number of linker dynamic rate unbinding forms must" <<
409  " match the number of species. Exiting." << endl;
410  return false;
411  }
412 
413  if(dy.dMUnbindingType.size() != sum(CParams.numMotorSpecies) &&
414  !dy.dMUnbindingType.empty()) {
415  cout << "Number of motor dynamic rate unbinding forms must" <<
416  " match the number of species. Exiting." << endl;
417  return false;
418  }
419  if(dy.dMWalkingType.size() != sum(CParams.numMotorSpecies) &&
420  !dy.dMWalkingType.empty()) {
421  cout << "Number of motor dynamic rate walking forms must" <<
422  " match the number of species. Exiting." << endl;
423  return false;
424  }
425 
426  //now check parameters
427  if(dy.dFPolymerizationType.size() != SysParams::DynamicRates().
428  dFilPolymerizationCharLength.size()) {
429  cout << "Must set a dynamic rate polymerization length for all filaments. Exiting." << endl;
430  return false;
431  }
432 
433  auto numCharLengths = 0;
434  auto numAmps = 0;
435 
436  for(auto &changer : dy.dLUnbindingType) {
437 
438  if(changer == "CATCHSLIP") {
439  numCharLengths += 2;
440  numAmps += 2;
441  }
442  else if(changer == "SLIP") {
443  numCharLengths += 1;
444  }
445 
446  }
447  if(numCharLengths != SysParams::DynamicRates().
448  dLinkerUnbindingCharLength.size()) {
449  cout << "Number of characteristic lengths specified for chosen "
450  << "linker unbinding dynamic rate forms is not accurate. Exiting."
451  << endl;
452  return false;
453  }
454 
455  if(numAmps != SysParams::DynamicRates().
456  dLinkerUnbindingAmplitude.size()) {
457 
458 
459  cout << "Number of amplitudes specified for chosen "
460  << "linker unbinding dynamic rate forms is not accurate. Exiting."
461  << endl;
462  return false;
463  }
464 
465  auto numCharForces = 0;
466 
467  for(auto &changer : dy.dMUnbindingType) {
468 
469  if(changer == "LOWDUTYCATCHSLIP") {
470  numCharForces += 2;
471  }
472  else if(changer == "LOWDUTYCATCH") {
473  numCharForces += 1;
474  }
475  else if(changer == "HIGHDUTYCATCH") {
476  numCharForces += 1;
477  }
478 
479  }
480  if(numCharForces != SysParams::DynamicRates().
481  dMotorUnbindingCharForce.size()) {
482  cout << "Number of characteristic forces specified for chosen "
483  << "motor unbinding dynamic rate forms is not accurate. Exiting."
484  << endl;
485  return false;
486  }
487 
488  if(dy.dMWalkingType.size() != SysParams::DynamicRates().
489  dMotorWalkingCharForce.size()) {
490  cout << "Number of characteristic forces specified for chosen "
491  << "motor walking dynamic rate forms is not accurate. Exiting."
492  << endl;
493  return false;
494  }
495 
496  return true;
497 }
