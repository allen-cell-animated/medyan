27 Compartment* GController::getCompartment(const vector<size_t> &indices)
28 {
29  size_t index = 0;
30  size_t i = 0;
31  for(auto x: indices)
32  {
33  //Flatten the indices to 1D
34  if(i == 0) {
35  if(x >= _grid[0])
36  throw OutOfBoundsException();
37 
38  index += x;
39  }
40  else if(i == 1) {
41  if(x >= _grid[1])
42  throw OutOfBoundsException();
43 
44  index += x * _grid[0];
45  }
46  else {
47  if(x >= _grid[2])
48  throw OutOfBoundsException();
49 
50  index += x * _grid[0] * _grid[1];
51  }
52 
53  i++;
54  }
55  try {
56  return _compartmentGrid->getCompartment(index);
57  }
58  catch (exception& e){
59  cout << "Bad compartment access at..." << endl;
60  cout << "Compartment index = " << index << endl;
61  cout << "Indices = " << indices[0] << " " << indices[1] << " " << indices[2] << endl;
62  throw NaNCoordinateException();
63  }
64 }

106 void GController::generateConnections()
107 {
108  for(size_t i=0U; i<_grid[0]; ++i) {
109 
110  for(size_t j=0U; j<_grid[1]; ++j) {
111 
112  for(size_t k=0U; k<_grid[2]; ++k)
113  {
114  vector<size_t> indices{i,j,k};
115  Compartment *target = getCompartment(indices);
116 
117  vector<double> coordinates =
118  {indices[0] * _compartmentSize[0] + _compartmentSize[0] / 2,
119  indices[1] * _compartmentSize[1] + _compartmentSize[1] / 2,
120  indices[2] * _compartmentSize[2] + _compartmentSize[2] / 2};
121  target->setCoordinates(coordinates);
122 
123  for(int ii: {-1,1})
124  {
125  int iprime = i+ii;
126  if(iprime<0 or iprime==int(_grid[0]))
127  continue;
128  vector<size_t> currentIndices{size_t(iprime), j, k};
129  Compartment *neighbor = getCompartment(currentIndices);
130  target->addNeighbour(neighbor);
131  }
132  for(int jj: {-1,1})
133  {
134  int jprime = j+jj;
135  if(jprime<0 or jprime==int(_grid[1]))
136  continue;
137  vector<size_t> currentIndices{i, size_t(jprime), k};
138  Compartment *neighbor = getCompartment(currentIndices);
139  target->addNeighbour(neighbor);
140  }
141  for(int kk: {-1,1})
142  {
143  int kprime = k+kk;
144  if(kprime<0 or kprime==int(_grid[2]))
145  continue;
146  vector<size_t> currentIndices{i, j, size_t(kprime)};
147  Compartment *neighbor = getCompartment(currentIndices);
148  target->addNeighbour(neighbor);
149  }
150  }
151  }
152  }

184  }
185  else {

187  exit(EXIT_FAILURE);
188  }
189 
190  int size = 1;

192  if(x != 0) size*=x;
193  }
194

206 
207  BoundaryType type;

230  move = BoundaryMove::All;

240

242  _boundary = new BoundaryCubic(_subSystem, move);

252 
253  _boundary = new BoundarySpherical(_subSystem,

255  }
256

258 
259  if(move != BoundaryMove::None) {
260 
261  cout << "Moving boundaries for a capsule shape "

284 void GController::findCompartments(const vector<double>& coords,
285  Compartment* ccheck, double dist,
286  vector<Compartment*>& compartments) {
287 
288  //base case : if c and ccheck are not within range, return
289  if(twoPointDistance(coords, ccheck->coordinates()) > dist ) return;
290 
291  //recursive case, c and ccheck are in range. call for all neighbors
292  else {
293  //if not already in list, add it
294  auto it = find(compartments.begin(), compartments.end(), ccheck);
295 
296  if( it == compartments.end()) {
297  //add the compartment
298  compartments.push_back(ccheck);
299 
300  //recursively call for all neighbors
301  for(auto &n : ccheck->getNeighbours())
302  findCompartments(coords, n, dist, compartments);
303  }
304  }
305 }
