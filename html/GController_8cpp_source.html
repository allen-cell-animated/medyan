84  //Three dimensional
85  if (_nDim == 3) {
86 
87  for(size_t i=0U; i<_grid[0]; ++i) {
88 
89  for(size_t j=0U; j<_grid[1]; ++j) {

92  {
93  vector<size_t> indices{i,j,k};

95 
96  vector<double> coordinates = {indices[0] * _compartmentSize[0] + _compartmentSize[0] / 2,

98  indices[2] * _compartmentSize[2] + _compartmentSize[2] / 2};

107  Compartment *neighbor = getCompartment(currentIndices);

243 void GController::findCompartments(const vector<double>& coords, Compartment* ccheck,
244  double dist, vector<Compartment*>& compartments) {
245 
246  //base case : if c and ccheck are not within range, return
247  if(TwoPointDistance(coords, ccheck->coordinates()) > dist ) return;
248 
249  //recursive case, c and ccheck are in range. call for all neighbors
250  else {
251  //if not already in list, add it
252  if(find(compartments.begin(), compartments.end(), ccheck) == compartments.end()) {
253  compartments.push_back(ccheck);
254  //recursively call for all neighbors
255  for(auto &n : ccheck->getNeighbours()) findCompartments(coords, n, dist, compartments);
256  }
257  }
