59

61

66

78  else inZone = true;

244  SysParams::Chemistry().linkerBoundIndex[_filamentType])->getN(), 1.0)) {
245 
246  //loop through neighbors
247  //now re add valid based on CCNL
248  for (auto cn : _neighborLists[_nlIndex]->getNeighbors(cc->getCylinder())) {

259

261  SysParams::Chemistry().linkerBoundIndex[_filamentType])->getN(), 1.0)) {

303  }
