68  static SpeciesNamesDB* Instance();
69 
71  string intToString(unsigned int i) const {
72  if (i>=_vec_int_string.size())
73  throw out_of_range("SpeciesNamesDB::intToString(int i) index error:[" + to_string(i) +"], while the vector size is " + to_string(_vec_int_string.size()));
74  return _vec_int_string[i];
75  }
76 
79  int stringToInt (string name) {
80  auto mit = _map_string_int.find(name);
81  if(mit == _map_string_int.end()){
82  _vec_int_string.push_back(name);
83  _map_string_int[name]=_vec_int_string.size()-1;
84  return _vec_int_string.size()-1;
85  }
86  else
87  return mit->second;
88  }
89 
91  string generateUniqueName(string name) {
92  string uniqueName = name + to_string(_num);
93  if(_map_string_int.find(uniqueName) != _map_string_int.end())
94  return uniqueName;
95  else
96  return name + to_string(++_num);
97  }
98 
100  void clear() {
101  _map_string_int.clear();
102  _vec_int_string.clear();
103  }
