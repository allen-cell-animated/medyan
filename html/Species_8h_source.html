58 class SpeciesNamesDB {

60 private:

62  static vector<string> _vec_int_string;
63 
64  static unsigned long _num;
65 public:
66 
69  static string intToString(unsigned int i) {
70 
71  if (i>=_vec_int_string.size())
72  throw out_of_range(
73  "SpeciesNamesDB::intToString(int i) index error:[" +
74  to_string(i) +"], while the vector size is " +
75  to_string(_vec_int_string.size()));
76 
77  return _vec_int_string[i];
78  }
79 
82  static int stringToInt (string name) {
83  auto mit = _map_string_int.find(name);
84 
85  if(mit == _map_string_int.end()){
86 
87  _vec_int_string.push_back(name);
88  _map_string_int[name]= _vec_int_string.size()-1;
89 
90  return _vec_int_string.size()-1;
91  }
92  else
93  return mit->second;
94  }
95 
99  static string genUniqueFilName(string name) {
100  string uniqueName = name + to_string(_num);
101  if(_map_string_int.find(uniqueName) != _map_string_int.end()) {
102 
103  return uniqueName;
104  }
105  else {
106  _vec_int_string.push_back(uniqueName);
107  _map_string_int[uniqueName] = _vec_int_string.size()-1;
108 
109  return name + "-" + to_string(++_num);
110  }
111  }
112 
115  static string removeUniqueFilName(string name) {
116 
117  //loop through string, get to integer
118  for(unsigned long i = 0; i < name.length(); i++) {
119 
120  if(name.substr(i,1).compare("-") == 0) {
121  //return the string, cutting before this value
122  return name.substr(0, i);
123  }
124  }
125  //if there was no integer, return the name
126  return name;
127  }
128 
132  static string genBindingName(string binding, string bound) {
133 
134  string name = binding + "-" + bound;
135 
136  _vec_int_string.push_back(name);
137  _map_string_int[name] = _vec_int_string.size()-1;
138 
139  return name;
140  }
141 
143  static void clear() {
144  _map_string_int.clear();
145  _vec_int_string.clear();
146  }
