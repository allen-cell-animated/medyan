74  vector<double> getAngleDeltaPos(vector<double>leg, vector<double> site1, vector<double> site2){
75  vector<double> returnVector;
76  double len1, len2, len3;
77  vector<double> v1={site1[0]-leg[0],site1[1]-leg[1],site1[2]-leg[2]};
78  vector<double> v2={site2[0]-leg[0],site2[1]-leg[1],site2[2]-leg[2]};
79  returnVector.push_back(1-acos(std::max(dotProduct(normalizedVector(v1),normalizedVector(v2)),-1.0))*7/22);
80  len1=twoPointDistance(leg,site1);
81  len2=twoPointDistance(leg,site2);

83  returnVector.push_back(abs(1-(len1+len2)/len3));
84  returnVector.push_back(len1);
85  returnVector.push_back(len3);
