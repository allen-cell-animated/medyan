85  void updateBindingReaction(int oldN, int newN) {
86 
87  int diff = newN - oldN;
88 
89  //update copy number
90  if(diff > 0) {
91  while (diff != 0) {
92  _bindingSpecies->up();
93  diff--;
94  }
95  }
96  else if(diff < 0) {
97  while (diff != 0) {
98  _bindingSpecies->down();
99  diff++;
100  }
101  }
102  else {} //do nothing
103 
104  //check if matching
105  assert((_bindingSpecies->getN() == numBindingSites())
106  && "Species representing binding sites does not match \
107  number of binding sites held by the manager.");
108 
109  _bindingReaction->updatePropensity();
110  }
