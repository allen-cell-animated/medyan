33  vector<tuple<string, short, vector<double>>> dummy2;

35  //Create random distribution of filaments
36  int filamentCounter = 0;
37  while (filamentCounter < numFilaments) {
38 
39  //Create a random filament vector one cylinder long
40  vector<double> firstPoint = GController::getRandomCoordinates();
41 
42  double directionX = Rand::randDouble(-1,1);
43  double directionY = Rand::randDouble(-1,1);
44  double directionZ = Rand::randDouble(-1,1);
45  vector<double> direction = normalizedVector({directionX, directionY, directionZ});
46 
47  vector<double> secondPoint =
48  nextPointProjection(firstPoint,(double)lenFilaments *
49  SysParams::Geometry().cylinderSize[filamentType] - 0.01, direction);
50

56  (twoPointDistance(bb->getBead()->coordinate, secondPoint) < bb->getRadius()))
57  inBubble = true;
58  }
59 
60  //check if within cutoff of boundary
61  bool outsideCutoff = false;
62  if(b->distance(firstPoint) < SysParams::Boundaries().BoundaryCutoff / 4.0 ||
63  b->distance(secondPoint) < SysParams::Boundaries().BoundaryCutoff / 4.0) {
64  outsideCutoff = true;
65  }
66 
67  if(b->within(firstPoint) && b->within(secondPoint) && !inBubble && !outsideCutoff) {
68  filaments.emplace_back(filamentType, firstPoint, secondPoint);
69  filamentCounter++;
70  }
71  }
72  return make_tuple(filaments, dummy, dummy2, dummy3);

128  directionZ = Rand::randDouble(-1,1);

143  SysParams::Geometry().cylinderSize[filamentType] - 0.01, randDirection);
144

147  b->distance(firstPoint) > safeDist &&
148  b->distance(secondPoint) > safeDist) {
149  filaments.emplace_back(filamentType, firstPoint, secondPoint);
150 
151  prevFirstPoint = firstPoint;
152  prevSecondPoint = secondPoint;
153 
154  filamentCounter++;
155  }

161  }
