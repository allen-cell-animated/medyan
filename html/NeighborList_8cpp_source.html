35 void CylinderNeighborList::updateNeighbors(Neighbor* n) {
36 
38  _list[n].clear();
39  Cylinder* cylinder = static_cast<Cylinder*>(n);
40 
42  vector<Compartment*> compartments;
43 
44  GController::findCompartments(cylinder->coordinate, cylinder->getCompartment(),
45  SystemParameters::Geometry().largestCompartmentSide * 2, compartments);
46 
47  for(auto &c : compartments) {
48  for(auto &nearbyCylinder : c->getCylinders()) {
49 
51  if(cylinder->getID() <= nearbyCylinder->getID()) continue;
52 
54  if(cylinder->getFilament() == nearbyCylinder->getFilament()) {
55 
57  if(_crossFilamentOnly) continue;
58 
60  else if(abs(cylinder->getPositionFilament()
61  - nearbyCylinder->getPositionFilament()) <= 2) continue;
62  }
63 
65  double dist = TwoPointDistance(cylinder->coordinate, nearbyCylinder->coordinate);
66  if(dist > _rMax || dist < _rMin) continue;
67 
69  _list[cylinder].push_back(nearbyCylinder);
70  }
71  }
