26 template <class BRepulsionInteractionType>
27 double BoundaryCylinderRepulsion<BRepulsionInteractionType>::computeEnergy(double d) {
28 
29  double U = 0;
30  double U_i;
31 
32  for (auto be: BoundaryElement::getBoundaryElements()) {
33 
34  for(auto &c : _neighborList->getNeighbors(be)) {
35 
36  double kRep = be->getRepulsionConst();
37  double screenLength = be->getScreeningLength();
38 
39  //potential acts on second bead unless this is a minus end
40  Bead* bd;
41  if(c->isMinusEnd())
42  bd = c->getFirstBead();
43  else
44  bd = c->getSecondBead();
45 
46  if (d == 0.0)
47  U_i = _FFType.energy(
48  bd, be->distance(bd->coordinate), kRep, screenLength);
49  else
50  U_i = _FFType.energy(
51  bd, be->stretchedDistance(bd->coordinate, bd->force, d), kRep, screenLength);
52 
53  if(fabs(U_i) == numeric_limits<double>::infinity()
54  || U_i != U_i || U_i < -1.0) {
55 
56  //set culprits and return
57  _otherCulprit = c;
58  _boundaryElementCulprit = be;
59 
60  return -1;
61  }
62  else
63  U += U_i;
64  }
65  }
66 
67  return U;
68 }
