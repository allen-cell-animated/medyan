185 void MotorGhost::updateReactionRates() {
186 
187  //current force
188  double force = max(0.0, _mMotorGhost->stretchForce);
189 
190  //update number of bound heads
191  if(!_unbindingChangers.empty())
192  _numBoundHeads = _unbindingChangers[_motorType]->numBoundHeads(_onRate, _offRate, force, _numHeads);
193  else
194  _numBoundHeads = _numHeads;
195 
196  //walking rate changer
197  if(!_walkingChangers.empty()) {
198 
199  auto x1 = _c1->getFirstBead()->coordinate;
200  auto x2 = _c1->getSecondBead()->coordinate;
201  auto x3 = _c2->getFirstBead()->coordinate;
202  auto x4 = _c2->getSecondBead()->coordinate;
203 
204  //get component of force in direction of forward walk for C1, C2
205  vector<double> motorC1Direction =
206  twoPointDirection(midPointCoordinate(x1, x2, _position1),
207  midPointCoordinate(x3, x4, _position2));
208 
209  vector<double> motorC2Direction =
210  twoPointDirection(midPointCoordinate(x3, x4, _position2),
211  midPointCoordinate(x1, x2, _position1));
212 
213  vector<double> c1Direction = twoPointDirection(x2,x1);
214  vector<double> c2Direction = twoPointDirection(x4,x3);
215 
216  double forceDotDirectionC1 = force * dotProduct(motorC1Direction, c1Direction);
217  double forceDotDirectionC2 = force * dotProduct(motorC2Direction, c2Direction);
218 
219  //WALKING REACTIONS
220  Species* s1 = _cMotorGhost->getFirstSpecies();
221  Species* s2 = _cMotorGhost->getSecondSpecies();
222 
223  for(auto r : s1->getRSpecies().reactantReactions()) {
224 
225  if(r->getReactionType() == ReactionType::MOTORWALKINGFORWARD) {
226 
227  float newRate =
228  _walkingChangers[_motorType]->
229  changeRate(_cMotorGhost->getOnRate(),
230  _cMotorGhost->getOffRate(),
231  _numHeads, max(0.0, forceDotDirectionC1));
232  if(SysParams::RUNSTATE==false){
233  newRate=0.0;}
234  r->setRate(newRate);
235  r->updatePropensity();
236 
237  }
238  else if(r->getReactionType() == ReactionType::MOTORWALKINGBACKWARD) {
239  float newRate =
240  _walkingChangers[_motorType]->
241  changeRate(_cMotorGhost->getOnRate(),
242  _cMotorGhost->getOffRate(),
243  _numHeads, max(0.0, -forceDotDirectionC1));
244 
245  if(SysParams::RUNSTATE==false){
246  newRate=0.0;}
247  r->setRate(newRate);
248 
249  r->updatePropensity();
250  }
251  }
252  for(auto r : s2->getRSpecies().reactantReactions()) {
253 
254  if(r->getReactionType() == ReactionType::MOTORWALKINGFORWARD) {
255 
256  float newRate =
257  _walkingChangers[_motorType]->
258  changeRate(_cMotorGhost->getOnRate(),
259  _cMotorGhost->getOffRate(),
260  _numHeads, max(0.0, forceDotDirectionC2));
261  if(SysParams::RUNSTATE==false)
262  { newRate=0.0;}
263 
264  r->setRate(newRate);
265  r->updatePropensity();
266  }
267  else if(r->getReactionType() == ReactionType::MOTORWALKINGFORWARD) {
268 
269  float newRate =
270  _walkingChangers[_motorType]->
271  changeRate(_cMotorGhost->getOnRate(),
272  _cMotorGhost->getOffRate(),
273  _numHeads, max(0.0, -forceDotDirectionC2));
274  if(SysParams::RUNSTATE==false)
275  { newRate=0.0;}
276  r->setRate(newRate);
277  r->updatePropensity();
278  }
279  }
280  }
281 
282  //unbinding rate changer
283  if(!_unbindingChangers.empty()) {
284 
285  //get the unbinding reaction
286  ReactionBase* offRxn = _cMotorGhost->getOffReaction();
287 
288  //change the rate
289  float newRate =
290  _unbindingChangers[_motorType]->
291  changeRate(_cMotorGhost->getOnRate(), _cMotorGhost->getOffRate(), _numHeads, force);
292 
293  offRxn->setRate(newRate);
294  offRxn->activateReaction();
295  }
