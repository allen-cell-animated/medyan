147 void Linker::updateReactionRates() {
148 
149  //if no rate changers were defined, skip
150  if(_unbindingChangers.empty()) return;
151 
152  //current force on linker
153  double force = max(0.0, _mLinker->stretchForce);
154 
155  //get the unbinding reaction
156  ReactionBase* offRxn = _cLinker->getOffReaction();
157 
158  //change the rate
159  float newRate = _unbindingChangers[_linkerType]->changeRate(offRxn->getBareRate(), force);
160  if(SysParams::RUNSTATE==false)
161  {newRate=0.0;}
162  offRxn->setRate(newRate);
163  offRxn->updatePropensity();
164 }
