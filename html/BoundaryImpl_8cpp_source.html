52 bool BoundaryCubic::within(Compartment* C) {
53 
54  for(auto &bs : _boundarySurfaces) {
55 
56  auto be = bs->boundaryElements()[0].get();
57 
58  //go half a compartment dist in the direction of normal
59  auto coordinate = C->coordinates();
60 
61  //initial check of coord
62  if(be->distance(coordinate) > 0) continue;
63 
64  //if not, see if any part is in bounds
65  auto normal = be->normal(coordinate);
66 
67  auto effCoordinate =
68  {coordinate[0] + SysParams::Geometry().compartmentSizeX * normal[0] / 2,
69  coordinate[1] + SysParams::Geometry().compartmentSizeY * normal[1] / 2,
70  coordinate[2] + SysParams::Geometry().compartmentSizeZ * normal[2] / 2};
71 
72  //check if this is within boundary
73  if(be->distance(effCoordinate) <= 0)
74  return false;
75 
76  }
77  return true;
78 }

136 void BoundaryCubic::move(double dist) {
137 
138  //do nothing
139  if(_move == BoundaryMove::None) return;
140 
141  //move the top plane
142  else if(_move == BoundaryMove::Top) {
143 
144  for(auto &bs : _boundarySurfaces) {
145 
146  auto be = bs->boundaryElements()[0].get();
147 
148  if(be->normal({0,0,0})[2] < 0) {
149 
150  be->updateCoords({be->_coords[0], be->_coords[1], be->_coords[2] + dist});
151  return;
152  }
153  }
154  }
155 
156  else if(_move == BoundaryMove::All) {
157 
158  for(auto &bs : _boundarySurfaces) {
159 
160  auto be = bs->boundaryElements()[0].get();
161 
162 
163  if(be->normal({0,0,0})[0] > 0)
164  be->updateCoords({be->_coords[0] - dist, be->_coords[1], be->_coords[2]});
165 
166  else if (be->normal({0,0,0})[0] < 0)
167  be->updateCoords({be->_coords[0] + dist, be->_coords[1], be->_coords[2]});
168 
169  else if (be->normal({0,0,0})[1] > 0)
170  be->updateCoords({be->_coords[0], be->_coords[1] - dist, be->_coords[2]});
171 
172  else if (be->normal({0,0,0})[1] < 0)
173  be->updateCoords({be->_coords[0], be->_coords[1] + dist, be->_coords[2]});
174 
175  else if (be->normal({0,0,0})[2] > 0)
176  be->updateCoords({be->_coords[0], be->_coords[1], be->_coords[2] - dist});
177 
178  else if (be->normal({0,0,0})[2] < 0)
179  be->updateCoords({be->_coords[0], be->_coords[1], be->_coords[2] + dist});
180 
181  }
182  }
183 }
