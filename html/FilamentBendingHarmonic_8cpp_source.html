50 
51  double L1 = sqrt(ScalarProduct(b1->coordinate, b2->coordinate, b1->coordinate, b2->coordinate));
52  double L2 = sqrt(ScalarProduct(b2->coordinate, b3->coordinate, b2->coordinate, b3->coordinate));
53  double l1l2 = ScalarProduct(b1->coordinate, b2->coordinate, b2->coordinate, b3->coordinate);
54 
55  //invL = 1/L;
56  double invL1 = 1/L1;
57  double invL2 = 1/L2;
58  double A = invL1*invL2;
59  double B = l1l2*invL1*A*A*L2;
60  double C = l1l2*invL2*A*A*L1;
61 
62  //force on i-1, f = k*(-A*l2 + B*l1):
63  b1->force[0] += k_bend * ( (-b3->coordinate[0] + b2->coordinate[0])*A + (b2->coordinate[0] - b1->coordinate[0])*B );
64  b1->force[1] += k_bend * ( (-b3->coordinate[1] + b2->coordinate[1])*A + (b2->coordinate[1] - b1->coordinate[1])*B );
65  b1->force[2] += k_bend * ( (-b3->coordinate[2] + b2->coordinate[2])*A + (b2->coordinate[2] - b1->coordinate[2])*B );
66 
67 
68  //force on i, f = k*(A*(l1-l2) - B*l1 + C*l2):
69  b2->force[0] += k_bend *( (b3->coordinate[0] - 2*b2->coordinate[0] + b1->coordinate[0])*A -
70  (b2->coordinate[0] - b1->coordinate[0])*B + (b3->coordinate[0] - b2->coordinate[0])*C );
71  b2->force[1] += k_bend *( (b3->coordinate[1] - 2*b2->coordinate[1] + b1->coordinate[1])*A -
72  (b2->coordinate[1] - b1->coordinate[1])*B + (b3->coordinate[1] - b2->coordinate[1])*C );
