134 TEST(ReactionTest, Dependents1) {
135  SpeciesBulk A("A", 10);
136  SpeciesBulk B("B", 10);
137  SpeciesBulk C("C", 10);
138  RSpecies& RA(A.getRSpecies());
139  Reaction<1,1> rxn1 = { {&A,&B}, 10.0 };
140  Reaction<1,1> rxn3 = { {&A,&C}, 10.0 };
141  Reaction<1,1> rxn2 = { {&B,&A}, 10.0 };
142 
143  rxn1.activateReaction();
144  rxn2.activateReaction();
145  rxn3.activateReaction();
146 
147  // note: we have three reactions, (1) A->B, (2) B->A, (3) A->C
148  // (1) affects (2) and (3)
149  // (2) affects (1) and (3)
150  // (3) affects (1)
151  vector<ReactionBase*> ar1 = rxn1.getAffectedReactions();
152  vector<ReactionBase*> ar2 = rxn2.getAffectedReactions();
153  vector<ReactionBase*> ar3 = rxn3.getAffectedReactions();
154  EXPECT_EQ(2U, ar1.size());
155  EXPECT_EQ(2U, ar2.size());
156  EXPECT_EQ(1U, ar3.size());
157  EXPECT_EQ(&rxn1, ar3[0]);// (3) affects (1)
158 
159  // Testing passivateAssocReacts() and activateAssocReactions()
160  // We run A->C 10 times, so A's copy number drops to 0. Hence, the reaction (1) & (3) should be passivated.
161  // In the context of this test, where we do not run a Gillespie-like algorithm, this should lead
162  // to reaction (2) not having (1) or (3) as dependents - the dependent count of (2) should become zero
163 
164  for (int i=0; i<10; ++i){
165  rxn3.makeStep();
166  }
167  EXPECT_EQ(0U, A.getN());
168 
169  EXPECT_EQ(0U, rxn2.dependents().size());
170 
171  // But not that the getAffectedReactions() still returns the original depedencices, ignoring the fact
172  // that copy number of A is zero.
173  ar1 = rxn1.getAffectedReactions();
174  ar2 = rxn2.getAffectedReactions();
175  ar3 = rxn3.getAffectedReactions();
176  EXPECT_EQ(2U, ar1.size());
177  EXPECT_EQ(2U, ar2.size());
178  EXPECT_EQ(1U, ar3.size());
179  EXPECT_EQ(&rxn1, ar3[0]);// (3) affects (1)
180 
181  // Now let's activate (1) and (3) by moving the copy number of A from 0 to 1
182  rxn2.makeStep();
183  EXPECT_EQ(1U, RA.getN());
184  EXPECT_EQ(2U, rxn2.dependents().size());
185 }
