236 TEST(ReactionTest, Dependents1) {
237  SpeciesBulk A("A", 10);
238  SpeciesBulk B("B", 10);
239  SpeciesBulk C("C", 10);
240  RSpecies& RA(A.getRSpecies());
241  Reaction<1,1> rxn1 = { {&A,&B}, 10.0 };
242  Reaction<1,1> rxn3 = { {&A,&C}, 10.0 };
243  Reaction<1,1> rxn2 = { {&B,&A}, 10.0 };
244 
245  rxn1.activateReaction();
246  rxn2.activateReaction();
247  rxn3.activateReaction();
248 
249  // note: we have three reactions, (1) A->B, (2) B->A, (3) A->C
250  // (1) affects (2) and (3)
251  // (2) affects (1) and (3)
252  // (3) affects (1)
253  vector<ReactionBase*> ar1 = rxn1.getAffectedReactions();
254  vector<ReactionBase*> ar2 = rxn2.getAffectedReactions();
255  vector<ReactionBase*> ar3 = rxn3.getAffectedReactions();
256  EXPECT_EQ(2U, ar1.size());
257  EXPECT_EQ(2U, ar2.size());
258  EXPECT_EQ(1U, ar3.size());
259  EXPECT_EQ(&rxn1, ar3[0]);// (3) affects (1)
260 
261  // Testing passivateAssocReacts() and activateAssocReactions()
262  // We run A->C 10 times, so A's copy number drops to 0.
263  // Hence, the reaction (1) & (3) should be passivated.
264  // In the context of this test, where we do not run a Gillespie-like algorithm,
265  // this should lead to reaction (2) not having (1) or (3) as dependents - the
266  // dependent count of (2) should become zero
267 
268  for (int i=0; i<10; ++i){
269  rxn3.makeStep();
270  }
271  EXPECT_EQ(0U, A.getN());
272 
273  EXPECT_EQ(0U, rxn2.dependents().size());
274 
275  // But not that the getAffectedReactions() still returns the original
276  // depedencices, ignoring the fact that copy number of A is zero.
277  ar1 = rxn1.getAffectedReactions();
278  ar2 = rxn2.getAffectedReactions();
279  ar3 = rxn3.getAffectedReactions();
280  EXPECT_EQ(2U, ar1.size());
281  EXPECT_EQ(2U, ar2.size());
282  EXPECT_EQ(1U, ar3.size());
283  EXPECT_EQ(&rxn1, ar3[0]);// (3) affects (1)
284 
285  // Now let's activate (1) and (3) by moving the copy number of A from 0 to 1
286  rxn2.makeStep();
287  EXPECT_EQ(1U, RA.getN());
288  EXPECT_EQ(2U, rxn2.dependents().size());
289 }
