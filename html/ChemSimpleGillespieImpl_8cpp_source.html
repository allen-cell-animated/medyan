43 bool ChemSimpleGillespieImpl::makeStep() {
44 
45  double a_total = computeTotalA();
46 
47  // this means that the network has come to a halt
48  if(a_total<1e-15)
49  return false;
50 
51  double tau = generateTau(a_total);
52  _t+=tau;
53  syncGlobalTime();
54 
55  //Gillespie algorithm's second step: finding which reaction happened;
56  double mu = a_total*generateUniform();
57  double rates_sum = 0;
58  ReactionBase* r_selected = nullptr;
59  for (auto &r : _reactions){
60  rates_sum+=r->computePropensity();
61  if(rates_sum>mu){
62  r_selected = r;
63  break;
64  }
65  }
66 
67  if(r_selected==nullptr){
68  cout << "ChemSimpleGillespieImpl::makeStep() for loop: rates_sum=" << rates_sum << ", mu="
69  << mu << ", a_total=" << a_total << endl;
70  throw runtime_error("ChemSimpleGillespieImpl::makeStep(): No Reaction was selected during the Gillespie step!");
71  }
72  r_selected->makeStep();
73 
74  // Send signals
75 #ifdef REACTION_SIGNALING
76  r_selected->emitSignal();
77 #endif
78 #ifdef RSPECIES_SIGNALING
79  r_selected->broadcastRSpeciesSignals();
80 #endif
81 
82  syncGlobalTime();
83  return true;
84 }
