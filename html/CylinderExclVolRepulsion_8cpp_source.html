42 double CylinderExclVolRepulsion::energy(Bead* b1, Bead* b2,
43  Bead* b3, Bead* b4,
44  double kRepuls) {
45 
46  auto c1 = b1->coordinate;
47  auto c2 = b2->coordinate;
48  auto c3 = b3->coordinate;
49  auto c4 = b4->coordinate;
50 
51  //check if parallel
52  if(areParallel(c1, c2, c3, c4)) {
53 
54  double d = twoPointDistance(c1, c3);
55  double invDSquare = 1 / (d * d);
56  double energy = kRepuls * invDSquare * invDSquare;
57 
58  return energy;
59  }
60 
61  //check if in same plane
62  if(areInPlane(c1, c2, c3, c4)) {
63 
64  //slightly move point
65  c2 = movePointOutOfPlane(c1, c2, c3, c4, 2, 0.01);
66  }
67 
68  double a = scalarProduct(c1, c2, c1, c2);
69  double b = scalarProduct(c3, c4, c3, c4);
70  double c = scalarProduct(c3, c1, c3, c1);
71  double d = scalarProduct(c1, c2, c3, c4);
72  double e = scalarProduct(c1, c2, c3, c1);
73  double f = scalarProduct(c3, c4, c3, c1);
74 
75  double AA = sqrt(a*c - e*e);
76  double BB = sqrt(b*c - f*f);
77 
78  double CC = d*e - a*f;
79  double DD = b*e - d*f;
80 
81  double EE = sqrt( a*(b + c - 2*f) - (d - e)*(d - e) );
82  double FF = sqrt( b*(a + c + 2*e) - (d + f)*(d + f) );
83 
84  double GG = d*d - a*b - CC;
85  double HH = CC + GG - DD;
86  double JJ = c*(GG + CC) + e*DD - f*CC;
87 
88 
89  double ATG1 = atan( (a + e)/AA) - atan(e/AA);
90  double ATG2 = atan((a + e - d)/EE) - atan((e - d)/EE);
91  double ATG3 = atan((f)/BB) - atan((f - b)/BB);
92  double ATG4 = atan((d + f)/FF) - atan((d + f - b)/FF);
93 
94  double energy = 0.5*kRepuls/JJ*( CC/AA*ATG1 + GG/EE*ATG2 + DD/BB*ATG3 + HH/FF*ATG4);
95 
96  return energy;
97 }
