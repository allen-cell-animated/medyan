50  double AA = sqrt(a*c - e*e);
51  double BB = sqrt(b*c - f*f);
52

55 
56  double EE = sqrt( a*(b + c - 2*f) - (d - e)*(d - e) );
57  double FF = sqrt( b*(a + c + 2*e) - (d + f)*(d + f) );

61  double HH = CC + GG - DD;
62  double JJ = c*(GG + CC) + e*DD - f*CC;
63 
64  double ATG1 = atan( (a + e)/AA) - atan(e/AA);
65  double ATG2 = atan((a + e - d)/EE) - atan((e - d)/EE);
66  double ATG3 = atan((f)/BB) - atan((f - b)/BB);
67  double ATG4 = atan((d + f)/FF) - atan((d + f - b)/FF);
68 
69  return 0.5*kRepuls/JJ*( CC/AA*ATG1 + GG/EE*ATG2 + DD/BB*ATG3 + HH/FF*ATG4 );
