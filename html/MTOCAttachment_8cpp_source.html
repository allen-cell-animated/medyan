43 
44  if (d == 0.0)
45  U_i = _FFType.energy(b1, b2, kStretch, radius);
46  else
47  U_i = _FFType.energy(b1, b2, kStretch, radius, d);
48 
49  if(fabs(U_i) == numeric_limits<double>::infinity()
50  || U_i != U_i || U_i < -1.0) {
51
