00063 void Reaction::activateReaction(){
00064 #ifdef TRACK_ZERO_COPY_N

00073     {
00074         for(auto r = (*s)->beginReactantReactions(); r!=(*s)->endReactantReactions(); ++r){
00075             if(this!=(*r))
00076                 (*r)->registerNewDependent(this);
00077         }
00078         for(auto r = (*s)->beginProductReactions(); r!=(*s)->endProductReactions(); ++r){
00079             if(this!=(*r))
00080                 (*r)->registerNewDependent(this);

00082     }
00083     _passivated=false;
00084     if(_rnode!=nullptr)
